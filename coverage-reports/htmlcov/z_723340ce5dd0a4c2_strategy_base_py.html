89class StrategyBase(ABC): 

90 """ 

91 戦略ベースクラス

97 def __init__(self, name: str, config: Optional[Dict[str, Any]] = None): 

98 """ 

99 戦略の初期化 

100 

101 Args: 

102 name: 戦略名 

103 config: 戦略設定辞書. 

104 """ 

105 self.name = name 

106 self.config = config or {} 

107 self.logger = get_logger() 

108 

109 # 戦略状態 

110 self.is_enabled = True 

111 self.last_signal: Optional[StrategySignal] = None 

112 self.signal_history: List[StrategySignal] = [] 

113 

114 # パフォーマンス追跡 

115 self.total_signals = 0 

116 self.successful_signals = 0 

117 self.last_update = datetime.now() 

118 

119 self.logger.info(f"戦略初期化完了: {self.name}")

121 @abstractmethod 

122 def analyze( 

123 self, df: pd.DataFrame, multi_timeframe_data: Optional[Dict[str, pd.DataFrame]] = None 

124 ) -> StrategySignal: 

125 """ 

126 市場データを分析してシグナルを生成（抽象メソッド） 

127 

128 Args: 

129 df: OHLCV + 特徴量データ（メインタイムフレーム） 

130 multi_timeframe_data: マルチタイムフレームデータ（Phase 31対応） 

131 - 形式: {"4h": DataFrame, "15m": DataFrame} 

132 - Noneの場合は単一タイムフレーム動作（後方互換） 

133 

134 Returns: 

135 生成されたシグナル 

136 

137 Raises:

152 def generate_signal( 

153 self, df: pd.DataFrame, multi_timeframe_data: Optional[Dict[str, pd.DataFrame]] = None 

154 ) -> StrategySignal: 

155 """ 

156 シグナル生成のメインエントリーポイント 

157 

158 Args: 

159 df: 市場データ 

160 multi_timeframe_data: マルチタイムフレームデータ（Phase 31対応） 

161 

162 Returns: 

163 生成されたシグナル. 

164 """ 

165 try: 

166 self.logger.debug(f"[{self.name}] シグナル生成開始") 

167 

168 # 前処理チェック 

169 self._validate_input_data(df) 

170 

171 # 戦略分析実行（Phase 31: multi_timeframe_data渡し） 

172 signal = self.analyze(df, multi_timeframe_data=multi_timeframe_data) 

173 

174 # 後処理 

175 self._post_process_signal(signal) 

176 

177 self.logger.debug( 

178 f"[{self.name}] シグナル生成完了: {signal.action} (信頼度: {signal.confidence:.3f})" 

179 ) 

180 

181 return signal 

182 

183 except Exception as e: 

184 self.logger.error(f"[{self.name}] シグナル生成エラー: {e}") 

185 raise StrategyError(f"戦略シグナル生成失敗: {e}", strategy_name=self.name) 

186 

187 def _validate_input_data(self, df: pd.DataFrame) -> None: 

188 """入力データの検証.""" 

189 if df.empty: 

190 raise StrategyError("データが空です", strategy_name=self.name) 

191 

192 # 必要特徴量の存在確認 

193 required_features = self.get_required_features() 

194 missing_features = [f for f in required_features if f not in df.columns] 

195 

196 if missing_features: 

197 raise StrategyError( 

198 f"必要特徴量が不足: {missing_features}", 

199 strategy_name=self.name, 

200 ) 

201 

202 # 最低データ数確認 

203 min_data_points = self.config.get("min_data_points", 20) 

204 if len(df) < min_data_points: 

205 raise StrategyError( 

206 f"データ数不足: {len(df)} < {min_data_points}", 

207 strategy_name=self.name, 

208 ) 

209 

210 def _post_process_signal(self, signal: StrategySignal) -> None: 

211 """シグナル後処理.""" 

212 # シグナル履歴に追加 

213 self.signal_history.append(signal) 

214 self.last_signal = signal 

215 self.total_signals += 1 

216 self.last_update = datetime.now() 

217 

218 # 履歴サイズ制限（メモリ使用量制御） 

219 max_history = self.config.get("max_signal_history", 1000) 

220 if len(self.signal_history) > max_history: 

221 self.signal_history = self.signal_history[-max_history:] 

222 

223 def update_performance(self, signal_success: bool) -> None: 

224 """パフォーマンス更新.""" 

225 if signal_success: 

226 self.successful_signals += 1 

227 

228 success_rate = self.get_success_rate() 

229 self.logger.info(f"[{self.name}] 成功率更新: {success_rate:.1f}%") 

230 

231 def get_success_rate(self) -> float: 

232 """成功率取得.""" 

233 if self.total_signals == 0: 

234 return 0.0 

235 return (self.successful_signals / self.total_signals) * 100 

236 

237 def get_signal_stats(self) -> Dict[str, Any]: 

238 """シグナル統計情報取得.""" 

239 if not self.signal_history: 

240 return {"total": 0, "by_action": {}, "avg_confidence": 0.0} 

241 

242 # アクション別カウント 

243 action_counts: Dict[str, int] = {} 

244 total_confidence: float = 0.0 

245 

246 for signal in self.signal_history: 

247 action = signal.action 

248 action_counts[action] = action_counts.get(action, 0) + 1 

249 total_confidence += signal.confidence 

250 

251 return { 

252 "total": len(self.signal_history), 

253 "by_action": action_counts, 

254 "avg_confidence": total_confidence / len(self.signal_history), 

255 "success_rate": self.get_success_rate(), 

256 "last_signal_time": ( 

257 self.last_signal.timestamp.isoformat() if self.last_signal else None 

258 ), 

259 }
