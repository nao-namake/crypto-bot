46 async def generate_backtest_report( 

47 self, final_stats: Dict[str, Any], start_date: datetime, end_date: datetime 

48 ) -> str: 

49 """ 

50 バックテストレポート生成（Phase 21） 

51 

52 Args: 

53 final_stats: バックテスト統計データ 

54 start_date: バックテスト開始日 

55 end_date: バックテスト終了日 

56 

57 Returns: 

58 生成されたレポートファイルパス 

59 """ 

60 self.logger.info("バックテストレポート生成開始") 

61 

62 timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") 

63 filename = f"backtest_{timestamp}.json" 

64 filepath = self.output_dir / filename 

65 

66 try: 

67 # レポートデータ構築 

68 report_data = { 

69 "backtest_info": { 

70 "start_date": start_date.isoformat(), 

71 "end_date": end_date.isoformat(), 

72 "duration_days": (end_date - start_date).days, 

73 "generated_at": datetime.now().isoformat(), 

74 "phase": "Phase_21_同一ロジック", 

75 }, 

76 "execution_stats": final_stats, 

77 "system_info": { 

78 "runner_type": "BacktestRunner", 

79 "data_source": "CSV", 

80 "logic_type": "本番同一ロジック", 

81 }, 

82 } 

83 

84 # JSONファイル保存 

85 with open(filepath, "w", encoding="utf-8") as f: 

86 json.dump(report_data, f, ensure_ascii=False, indent=2) 

87 

88 self.logger.info(f"バックテストレポート生成完了: {filepath}") 

89 return str(filepath) 

90 

91 except Exception as e: 

92 self.logger.error(f"レポート生成エラー: {e}") 

93 raise
