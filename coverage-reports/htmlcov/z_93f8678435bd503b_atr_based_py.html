86 def _analyze_bb_position(self, df: pd.DataFrame) -> Dict[str, Any]: 

87 """ボリンジャーバンド位置分析.""" 

88 try: 

89 current_bb_pos = float(df["bb_position"].iloc[-1]) 

90 

91 # BB位置によるシグナル判定（逆張り） 

92 if current_bb_pos >= self.config["bb_overbought"]: 

93 signal = -1 # 売りシグナル（過買い） 

94 strength = min((current_bb_pos - self.config["bb_overbought"]) / 0.2, 1.0) 

95 elif current_bb_pos <= self.config["bb_oversold"]: 

96 signal = 1 # 買いシグナル（過売り） 

97 strength = min((self.config["bb_oversold"] - current_bb_pos) / 0.2, 1.0) 

98 else: 

99 signal = 0 # ニュートラル 

100 strength = 0.0 

101 

102 confidence = 0.3 + strength * 0.4 if abs(signal) > 0 else 0.0 

103 

104 return { 

105 "signal": signal, 

106 "strength": strength, 

107 "confidence": confidence, 

108 "bb_position": current_bb_pos, 

109 "analysis": f"BB位置: {current_bb_pos:.2f} -> {['売り', 'なし', '買い'][signal + 1]}", 

110 } 

111 

112 except Exception as e: 

113 self.logger.error(f"BB位置分析エラー: {e}") 

114 return { 

115 "signal": 0, 

116 "strength": 0.0, 

117 "confidence": 0.0, 

118 "analysis": "エラー", 

119 }

156 def _analyze_atr_volatility(self, df: pd.DataFrame) -> Dict[str, Any]: 

157 """ATRボラティリティ分析.""" 

158 try: 

159 current_atr = float(df["atr_14"].iloc[-1]) 

160 current_price = float(df["close"].iloc[-1]) 

161 atr_ratio = current_atr / current_price 

162 

163 # 過去20期間の平均ATR比率と比較 

164 if len(df) >= 20: 

165 historical_atr = df["atr_14"].iloc[-20:] / df["close"].iloc[-20:] 

166 avg_atr_ratio = historical_atr.mean() 

167 volatility_multiplier = atr_ratio / (avg_atr_ratio + 1e-8) 

168 else: 

169 volatility_multiplier = 1.0 

170 

171 # ボラティリティ状態判定 

172 if volatility_multiplier < self.config["min_atr_ratio"]: 

173 regime = "low" # 低ボラ（取引回避） 

174 strength = 0.0 

175 elif volatility_multiplier > 1.5: 

176 regime = "high" # 高ボラ（逆張り好機） 

177 strength = min((volatility_multiplier - 1.5) / 0.5, 1.0) 

178 else: 

179 regime = "normal" # 通常ボラ 

180 strength = 0.5 

181 

182 return { 

183 "regime": regime, 

184 "strength": strength, 

185 "atr_ratio": atr_ratio, 

186 "volatility_multiplier": volatility_multiplier, 

187 "analysis": f"ボラティリティ: {regime} (倍率: {volatility_multiplier:.2f})", 

188 } 

189 

190 except Exception as e: 

191 self.logger.error(f"ATR分析エラー: {e}") 

192 return {"regime": "normal", "strength": 0.0, "analysis": "エラー"}

218 def _make_decision( 

219 self, 

220 bb_analysis: Dict[str, Any], 

221 rsi_analysis: Dict[str, Any], 

222 atr_analysis: Dict[str, Any], 

223 stress_analysis: Dict[str, Any] = None, 

224 ) -> Dict[str, Any]: 

225 """統合判定.""" 

226 try: 

227 # フィルター確認（Phase 19: market_stress無効化） 

228 if stress_analysis and not stress_analysis["filter_ok"]: 

229 return self._create_hold_decision("市場ストレス高") 

230 

231 if atr_analysis["regime"] == "low": 

232 return self._create_hold_decision("低ボラティリティ") 

233 

234 # シグナル統合 

235 bb_signal = bb_analysis["signal"] 

236 rsi_signal = rsi_analysis["signal"] 

237 

238 # 一致度確認 

239 if bb_signal != 0 and rsi_signal != 0: 

240 if bb_signal == rsi_signal: 

241 # 両方一致 

242 action = EntryAction.BUY if bb_signal > 0 else EntryAction.SELL 

243 confidence = min( 

244 bb_analysis["confidence"] + rsi_analysis["confidence"], 

245 1.0, 

246 ) 

247 strength = (bb_analysis["strength"] + rsi_analysis["strength"]) / 2 

248 else: 

249 # 不一致はホールド 

250 return self._create_hold_decision("シグナル不一致") 

251 elif bb_signal != 0: 

252 # BBシグナルのみ 

253 action = EntryAction.BUY if bb_signal > 0 else EntryAction.SELL 

254 confidence = bb_analysis["confidence"] * 0.7 # 単一シグナルなので減額 

255 strength = bb_analysis["strength"] 

256 elif rsi_signal != 0: 

257 # RSIシグナルのみ 

258 action = EntryAction.BUY if rsi_signal > 0 else EntryAction.SELL 

259 confidence = rsi_analysis["confidence"] * 0.7 

260 strength = rsi_analysis["strength"] 

261 else: 

262 return self._create_hold_decision("シグナルなし") 

263 

264 # 高ボラティリティボーナス 

265 if atr_analysis["regime"] == "high": 

266 confidence = min(confidence * 1.2, 1.0) 

267 

268 # 最小信頼度チェック 

269 if confidence < self.config["min_confidence"]: 

270 return self._create_hold_decision("信頼度不足") 

271 

272 return { 

273 "action": action, 

274 "confidence": confidence, 

275 "strength": strength, 

276 "analysis": f"ATR逆張り: {action} (信頼度: {confidence:.3f})", 

277 } 

278 

279 except Exception as e: 

280 self.logger.error(f"統合判定エラー: {e}") 

281 return self._create_hold_decision("判定エラー")
