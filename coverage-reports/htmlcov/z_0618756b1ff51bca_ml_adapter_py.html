65 MLServiceAdapter初期化

67 Args: 

68 logger: ログシステム 

69 """ 

70 self.logger = logger 

71 self.model = None

78 def _load_model(self) -> None: 

79 """ 

80 優先順位付きモデル読み込み 

81 

82 1. ProductionEnsemble（最優先） 

83 2. 個別モデルから再構築 

84 3. ダミーモデル（最終フォールバック） 

85 """ 

86 self.logger.info("🤖 MLモデル読み込み開始") 

87 

88 # 1. ProductionEnsemble読み込み試行 

89 if self._load_production_ensemble(): 

90 return 

91 

92 # 2. 個別モデルから再構築試行 

93 if self._load_from_individual_models(): 

94 return 

95 

96 # 3. 最終フォールバック: ダミーモデル 

97 self._load_dummy_model()

99 def _load_production_ensemble(self) -> bool: 

100 """ProductionEnsemble読み込み（互換性レイヤー付き）""" 

101 model_path = Path("models/production/production_ensemble.pkl") 

102 

103 if not model_path.exists(): 

104 self.logger.warning(f"ProductionEnsemble未発見: {model_path}") 

105 return False 

106 

107 try: 

108 # 🚨 CRITICAL FIX: 古いimportパス互換性レイヤー 

109 import sys 

110 

111 # pickle化されたモデル内の古いimportパスをリダイレクト 

112 class ModuleRedirect: 

113 def __getattr__(self, name): 

114 if name == "ProductionEnsemble": 

115 from ..ml.ensemble.production_ensemble import ProductionEnsemble 

116 

117 return ProductionEnsemble 

118 raise AttributeError(f"Module {name} not found") 

119 

120 # 一時的に古いパスをリダイレクト 

121 old_module = sys.modules.get("src.ml.production") 

122 sys.modules["src.ml.production"] = ModuleRedirect() 

123 

124 try: 

125 with open(model_path, "rb") as f: 

126 self.model = pickle.load(f) 

127 finally: 

128 # リダイレクト後片付け 

129 if old_module is None: 

130 sys.modules.pop("src.ml.production", None) 

131 else: 

132 sys.modules["src.ml.production"] = old_module 

133 

134 # モデルの妥当性チェック 

135 if hasattr(self.model, "predict") and hasattr(self.model, "predict_proba"): 

136 self.model_type = "ProductionEnsemble" 

137 self.is_fitted = getattr(self.model, "is_fitted", True) 

138 

139 self.logger.info("✅ ProductionEnsemble読み込み成功") 

140 model_info = ( 

141 self.model.get_model_info() if hasattr(self.model, "get_model_info") else "N/A" 

142 ) 

143 self.logger.info(f"モデル詳細: {model_info}") 

144 return True 

145 else: 

146 self.logger.error("ProductionEnsembleに必須メソッドが不足") 

147 return False 

148 

149 except Exception as e: 

150 self.logger.error(f"ProductionEnsemble読み込みエラー: {e}") 

151 return False

153 def _load_from_individual_models(self) -> bool: 

154 """個別モデルからProductionEnsemble再構築""" 

155 training_path = Path("models/training") 

156 

157 if not training_path.exists(): 

158 self.logger.warning(f"個別モデルディレクトリ未発見: {training_path}") 

159 return False 

160 

161 try: 

162 individual_models = {} 

163 

164 # 必要なモデルファイルの確認 

165 model_files = { 

166 "lightgbm": "lightgbm_model.pkl", 

167 "xgboost": "xgboost_model.pkl", 

168 "random_forest": "random_forest_model.pkl", 

169 } 

170 

171 for model_name, filename in model_files.items(): 

172 model_file = training_path / filename 

173 if model_file.exists(): 

174 with open(model_file, "rb") as f: 

175 individual_models[model_name] = pickle.load(f) 

176 self.logger.info(f"個別モデル読み込み: {model_name}") 

177 else: 

178 self.logger.warning(f"個別モデル未発見: {model_file}") 

179 

180 if len(individual_models) > 0: 

181 # ProductionEnsembleを再構築 

182 from ..ml.ensemble.production_ensemble import ProductionEnsemble 

183 

184 self.model = ProductionEnsemble(individual_models) 

185 self.model_type = "ReconstructedEnsemble" 

186 self.is_fitted = True 

187 

188 self.logger.info( 

189 f"✅ 個別モデルからEnsemble再構築成功 ({len(individual_models)}モデル)" 

190 ) 

191 return True 

192 else: 

193 self.logger.error("有効な個別モデルが見つかりません") 

194 return False 

195 

196 except Exception as e: 

197 self.logger.error(f"個別モデルからの再構築エラー: {e}") 

198 return False

209 def predict( 

210 self, X: Union[pd.DataFrame, np.ndarray], use_confidence: bool = True 

211 ) -> np.ndarray: 

212 """ 

213 統一predict インターフェース 

214 

215 Args: 

216 X: 特徴量データ 

217 use_confidence: 信頼度閾値使用（EnsembleModelのみ） 

218 

219 Returns: 

220 np.ndarray: 予測結果 

221 """ 

222 if not self.is_fitted: 

223 raise ValueError("モデルが学習されていません") 

224 

225 try: 

226 # EnsembleModelの場合は use_confidence パラメータを渡す 

227 if ( 

228 hasattr(self.model, "predict") 

229 and "use_confidence" in self.model.predict.__code__.co_varnames 

230 ): 

231 return self.model.predict(X, use_confidence=use_confidence) 

232 else: 

233 return self.model.predict(X) 

234 

235 except Exception as e: 

236 self.logger.error(f"予測エラー: {e}") 

237 

238 # エラー時はダミーモデルにフォールバック 

239 if self.model_type != "DummyModel": 

240 self.logger.warning("予測エラーによりダミーモデルにフォールバック") 

241 self._load_dummy_model() 

242 return self.model.predict(X) 

243 else: 

244 raise

276 "model_type": self.model_type,

284 base_info.update(model_info)
