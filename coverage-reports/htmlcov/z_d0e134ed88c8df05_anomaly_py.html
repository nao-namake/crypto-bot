87 def comprehensive_anomaly_check( 

88 self, 

89 bid: float, 

90 ask: float, 

91 last_price: float, 

92 volume: float, 

93 api_latency_ms: float, 

94 market_data: pd.DataFrame, 

95 ) -> List[AnomalyAlert]: 

96 """ 

97 包括的異常チェック 

98 

99 Args: 

100 bid: 買い価格 

101 ask: 売り価格 

102 last_price: 最終取引価格 

103 volume: 出来高 

104 api_latency_ms: API応答時間（ミリ秒） 

105 market_data: 市場データ履歴 

106 

107 Returns: 

108 検出された異常アラートリスト 

109 """ 

110 alerts = [] 

111 

112 # 1. スプレッド異常チェック 

113 spread_alert = self._check_spread_anomaly(bid, ask, last_price) 

114 if spread_alert: 

115 alerts.append(spread_alert) 

116 

117 # 2. API遅延チェック 

118 latency_alert = self._check_api_latency(api_latency_ms) 

119 if latency_alert: 

120 alerts.append(latency_alert) 

121 

122 # 3. 価格スパイクチェック 

123 price_alert = self._check_price_spike(last_price, market_data) 

124 if price_alert: 

125 alerts.append(price_alert) 

126 

127 # 4. ボリューム異常チェック 

128 volume_alert = self._check_volume_anomaly(volume, market_data) 

129 if volume_alert: 

130 alerts.append(volume_alert) 

131 

132 # 履歴に追加 

133 for alert in alerts: 

134 self.anomaly_history.append(alert) 

135 

136 # 古い履歴削除（24時間以上前） 

137 cutoff_time = datetime.now() - timedelta(hours=24) 

138 self.anomaly_history = [a for a in self.anomaly_history if a.timestamp >= cutoff_time] 

139 

140 return alerts

142 def _check_spread_anomaly( 

143 self, bid: float, ask: float, last_price: float 

144 ) -> Optional[AnomalyAlert]: 

145 """スプレッド異常チェック""" 

146 if last_price <= 0: 

147 return None 

148 

149 spread_pct = (ask - bid) / last_price 

150 

151 if spread_pct >= self.spread_critical_threshold: 

152 return AnomalyAlert( 

153 timestamp=datetime.now(), 

154 level=AnomalyLevel.CRITICAL, 

155 message=f"危険なスプレッド: {spread_pct:.2%}", 

156 details={"bid": bid, "ask": ask, "spread_pct": spread_pct}, 

157 ) 

158 elif spread_pct >= self.spread_warning_threshold: 

159 return AnomalyAlert( 

160 timestamp=datetime.now(), 

161 level=AnomalyLevel.WARNING, 

162 message=f"広いスプレッド: {spread_pct:.2%}", 

163 details={"bid": bid, "ask": ask, "spread_pct": spread_pct}, 

164 ) 

165 

166 return None

168 def _check_api_latency(self, api_latency_ms: float) -> Optional[AnomalyAlert]: 

169 """API遅延チェック""" 

170 if api_latency_ms >= self.api_latency_critical_ms: 

171 return AnomalyAlert( 

172 timestamp=datetime.now(), 

173 level=AnomalyLevel.CRITICAL, 

174 message=f"重大なAPI遅延: {api_latency_ms:.0f}ms", 

175 details={"latency_ms": api_latency_ms}, 

176 ) 

177 elif api_latency_ms >= self.api_latency_warning_ms: 

178 return AnomalyAlert( 

179 timestamp=datetime.now(), 

180 level=AnomalyLevel.WARNING, 

181 message=f"API遅延: {api_latency_ms:.0f}ms", 

182 details={"latency_ms": api_latency_ms}, 

183 ) 

184 

185 return None

187 def _check_price_spike( 

188 self, last_price: float, market_data: pd.DataFrame 

189 ) -> Optional[AnomalyAlert]: 

190 """価格スパイクチェック""" 

191 try: 

192 if len(market_data) < 20: 

193 return None 

194 

195 recent_prices = market_data["close"].tail(20) 

196 price_mean = recent_prices.mean() 

197 price_std = recent_prices.std() 

198 

199 if price_std > 0: 

200 z_score = abs((last_price - price_mean) / price_std) 

201 

202 if z_score >= self.price_spike_zscore_threshold: 

203 return AnomalyAlert( 

204 timestamp=datetime.now(), 

205 level=AnomalyLevel.CRITICAL, 

206 message=f"価格急変検出: Zスコア={z_score:.2f}", 

207 details={ 

208 "last_price": last_price, 

209 "mean_price": price_mean, 

210 "z_score": z_score, 

211 }, 

212 ) 

213 except Exception as e: 

214 self.logger.error(f"価格スパイクチェックエラー: {e}") 

215 

216 return None 

217 

218 def _check_volume_anomaly( 

219 self, volume: float, market_data: pd.DataFrame 

220 ) -> Optional[AnomalyAlert]: 

221 """ボリューム異常チェック""" 

222 try: 

223 if len(market_data) < 20 or "volume" not in market_data.columns: 

224 return None 

225 

226 recent_volumes = market_data["volume"].tail(20) 

227 volume_mean = recent_volumes.mean() 

228 volume_std = recent_volumes.std() 

229 

230 if volume_std > 0: 

231 z_score = abs((volume - volume_mean) / volume_std) 

232 

233 if z_score >= self.volume_spike_zscore_threshold: 

234 return AnomalyAlert( 

235 timestamp=datetime.now(), 

236 level=AnomalyLevel.WARNING, 

237 message=f"ボリューム異常: Zスコア={z_score:.2f}", 

238 details={ 

239 "current_volume": volume, 

240 "mean_volume": volume_mean, 

241 "z_score": z_score, 

242 }, 

243 ) 

244 except Exception as e: 

245 self.logger.error(f"ボリューム異常チェックエラー: {e}") 

246 

247 return None
