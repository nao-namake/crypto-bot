1""" 

2Graceful Shutdown管理 - GracefulShutdownManager 

3 

4main.py軽量化方針に従い、shutdown処理を専用サービスとして分離。 

5設計原則： 

6- Single Responsibility: shutdown処理のみ担当 

7- 依存性注入: orchestratorを外部から受け取り 

8- エラーハンドリング: 適切なログ出力・例外処理 

9- 統一設定管理体系: thresholds.yaml準拠 

10""" 

11 

12import asyncio 

13import signal 

14import sys 

15from typing import Optional 

16 

17from ..logger import CryptoBotLogger 

18 

19 

20class GracefulShutdownManager: 

21 """ 

22 Graceful Shutdown管理サービス 

23 

24 機能: 

25 - シグナルハンドリング設定 

26 - オーケストレーターとランナーの正常終了 

27 - タイムアウト管理 

28 - shutdown_event による協調的終了 

29 """ 

30 

31 def __init__(self, logger: CryptoBotLogger): 

32 """ 

33 初期化 

34 

35 Args: 

36 logger: CryptoBotLogger インスタンス 

37 """ 

38 self.logger = logger 

39 self.orchestrator: Optional[object] = None 

40 self.shutdown_event: Optional[asyncio.Event] = None 

41 self._shutdown_timeout = 30 # 30秒タイムアウト 

42 

43 def initialize(self, orchestrator: object) -> None: 

44 """ 

45 shutdown管理初期化 

46 

47 Args: 

48 orchestrator: TradingOrchestrator インスタンス 

49 """ 

50 self.orchestrator = orchestrator 

51 self.shutdown_event = asyncio.Event() 

52 self._setup_signal_handlers() 

53 self.logger.info("✅ Graceful shutdown管理初期化完了") 

54 

55 def _setup_signal_handlers(self) -> None: 

56 """ 

57 シグナルハンドリング設定（graceful shutdown対応） 

58 """ 

59 

60 def signal_handler(signum, frame): 

61 signal_name = signal.Signals(signum).name 

62 self.logger.info(f"🛑 シグナル受信: {signal_name} - graceful shutdown開始") 

63 

64 # shutdown_eventをセットしてメインループに終了を通知 

65 if self.shutdown_event: 

66 self.shutdown_event.set() 

67 else: 

68 # shutdown_event未設定の場合は即座に終了 

69 self.logger.warning("⚠️ shutdown_event未設定 - 即座に終了") 

70 sys.exit(0) 

71 

72 # SIGINT（Ctrl+C）とSIGTERM（kill）の処理 

73 signal.signal(signal.SIGINT, signal_handler) 

74 signal.signal(signal.SIGTERM, signal_handler) 

75 

76 async def wait_for_shutdown_signal(self) -> None: 

77 """ 

78 shutdown シグナル待機 

79 

80 Returns: 

81 shutdown_event がセットされるまで待機 

82 """ 

83 if self.shutdown_event: 

84 await self.shutdown_event.wait() 

85 else: 

86 # shutdown_event未設定の場合は無限待機 

87 await asyncio.Event().wait() 

88 

89 async def perform_graceful_shutdown(self) -> None: 

90 """ 

91 Graceful shutdown処理実行 

92 

93 オーケストレーターとランナーのクリーンアップを適切な順序で実行 

94 タイムアウト機能付き 

95 """ 

96 self.logger.info("🔄 Graceful shutdown処理開始") 

97 

98 try: 

99 # タイムアウト付きでshutdown処理実行 

100 await asyncio.wait_for(self._perform_cleanup(), timeout=self._shutdown_timeout) 

101 

102 self.logger.info("✅ Graceful shutdown完了") 

103 

104 except asyncio.TimeoutError: 

105 self.logger.error(f"❌ Graceful shutdown タイムアウト（{self._shutdown_timeout}秒）") 

106 except Exception as e: 

107 self.logger.error(f"❌ Graceful shutdown中にエラー: {e}") 

108 

109 async def _perform_cleanup(self) -> None: 

110 """ 

111 クリーンアップ処理実行（内部メソッド） 

112 """ 

113 if not self.orchestrator: 

114 self.logger.warning("⚠️ orchestrator未設定 - cleanup処理スキップ") 

115 return 

116 

117 cleanup_tasks = [] 

118 

119 # 各ランナーのクリーンアップタスクを並行実行 

120 if ( 

121 hasattr(self.orchestrator, "paper_trading_runner") 

122 and self.orchestrator.paper_trading_runner 

123 ): 

124 cleanup_tasks.append(self.orchestrator.paper_trading_runner.cleanup_mode()) 

125 

126 if ( 

127 hasattr(self.orchestrator, "live_trading_runner") 

128 and self.orchestrator.live_trading_runner 

129 ): 

130 cleanup_tasks.append(self.orchestrator.live_trading_runner.cleanup_mode()) 

131 

132 if hasattr(self.orchestrator, "backtest_runner") and self.orchestrator.backtest_runner: 

133 cleanup_tasks.append(self.orchestrator.backtest_runner.cleanup_mode()) 

134 

135 if cleanup_tasks: 

136 # 全クリーンアップタスクを並行実行 

137 await asyncio.gather(*cleanup_tasks, return_exceptions=True) 

138 self.logger.info("✅ 全ランナーのクリーンアップ完了") 

139 else: 

140 self.logger.info("⚠️ クリーンアップ対象ランナーが見つかりません") 

141 

142 async def shutdown_with_main_task(self, main_task: asyncio.Task) -> None: 

143 """ 

144 メインタスクと連携したshutdown処理 

145 

146 Args: 

147 main_task: メイン処理のasyncio.Task 

148 """ 

149 # shutdown監視タスク作成 

150 shutdown_task = asyncio.create_task(self.wait_for_shutdown_signal()) 

151 

152 # いずれかが完了するまで待機 

153 done, pending = await asyncio.wait( 

154 [main_task, shutdown_task], return_when=asyncio.FIRST_COMPLETED 

155 ) 

156 

157 # shutdown_eventがセットされた場合はgraceful shutdown実行 

158 if shutdown_task in done: 

159 self.logger.info("シグナル受信 - graceful shutdown開始") 

160 

161 # メイン処理をキャンセル 

162 main_task.cancel() 

163 try: 

164 await main_task 

165 except asyncio.CancelledError: 

166 self.logger.info("メイン処理のキャンセル完了") 

167 

168 # graceful shutdown実行 

169 await self.perform_graceful_shutdown() 

170 

171 # 残りのタスクをキャンセル 

172 for task in pending: 

173 task.cancel() 

174 try: 

175 await task 

176 except asyncio.CancelledError: 

177 pass
