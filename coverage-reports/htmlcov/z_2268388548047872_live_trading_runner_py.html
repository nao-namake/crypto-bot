33 """ 

34 ライブトレードモード実行 

35 

36 Returns: 

37 実行成功・失敗 

38 """ 

39 try: 

40 self.logger.info("🚨 ライブトレードモード開始", discord_notify=True) 

41 

42 # セッション開始 

43 self.session_start = datetime.now() 

44 self.cycle_count = 0 

45

58 await self._generate_final_summary()

60 raise

64 await self._save_error_report(str(e))

67 async def _validate_live_conditions(self) -> bool: 

68 """ライブトレード実行条件確認""" 

69 try: 

70 # 基本条件確認 

71 if self.config.mode != "live": 

72 self.logger.error("❌ 設定モードがライブトレードではありません") 

73 return False 

74 

75 # 必須サービス確認 

76 required_services = [ 

77 self.orchestrator.execution_service, 

78 self.orchestrator.risk_service, 

79 self.orchestrator.ml_service, 

80 ] 

81 

82 for service in required_services: 

83 if service is None: 

84 self.logger.error("❌ 必須サービスが初期化されていません") 

85 return False 

86 

87 # 残高確認（最小残高チェック・条件緩和） 

88 try: 

89 balance = await self._check_account_balance() 

90 min_balance = get_threshold( 

91 "trading.minimum_live_balance", 5000 

92 ) # 5千円最小（緩和） 

93 

94 if balance < min_balance: 

95 self.logger.warning( 

96 f"⚠️ 残高不足: {balance:,.0f}円 < {min_balance:,.0f}円（監視モードで継続）" 

97 ) 

98 # 以前は return False でシステム停止していたが、継続するように変更 

99 # 監視モードで継続し、システムを停止させない 

100 else: 

101 self.logger.info(f"✅ 残高確認完了: {balance:,.0f}円") 

102 

103 except Exception as e: 

104 self.logger.warning(f"⚠️ 残高確認エラー（継続稼働）: {e}") 

105 # エラーが発生してもシステム停止させない 

106 

107 self.logger.info("✅ ライブトレード実行条件確認完了") 

108 return True 

109 

110 except Exception as e: 

111 self.logger.error(f"❌ ライブトレード条件確認エラー: {e}") 

112 return False

114 async def _check_account_balance(self) -> float: 

115 """アカウント残高確認""" 

116 try: 

117 # 実行サービスから残高取得（実装は execution_service に依存） 

118 balance = getattr(self.orchestrator.execution_service, "current_balance", 0) 

119 

120 if balance <= 0: 

121 # 残高が0以下の場合は API から再取得を試行 

122 self.logger.warning("⚠️ 残高が0以下のため、再取得を試行します") 

123 

124 # 実際の残高再取得処理を実装 

125 try: 

126 from ...data.bitbank_client import BitbankClient 

127 from ..config import get_threshold 

128 

129 client = BitbankClient() 

130 balance_data = client.fetch_balance() 

131 jpy_balance = balance_data.get("JPY", {}).get("free", 0.0) 

132 

133 if jpy_balance > 0: 

134 # execution_serviceの残高を更新 

135 if hasattr(self.orchestrator.execution_service, "current_balance"): 

136 self.orchestrator.execution_service.current_balance = jpy_balance 

137 balance = jpy_balance 

138 self.logger.info(f"✅ 残高再取得成功: {jpy_balance:,.0f}円") 

139 else: 

140 # フォールバック値使用 

141 fallback = get_threshold("trading.initial_balance_jpy", 10000.0) 

142 if hasattr(self.orchestrator.execution_service, "current_balance"): 

143 self.orchestrator.execution_service.current_balance = fallback 

144 balance = fallback 

145 self.logger.warning(f"⚠️ 残高0のためフォールバック使用: {fallback:,.0f}円") 

146 

147 except Exception as re_error: 

148 self.logger.error(f"❌ 残高再取得失敗: {re_error}") 

149 # エラー時もフォールバック使用 

150 fallback = get_threshold("trading.initial_balance_jpy", 10000.0) 

151 if hasattr(self.orchestrator.execution_service, "current_balance"): 

152 self.orchestrator.execution_service.current_balance = fallback 

153 balance = fallback 

154 self.logger.warning( 

155 f"⚠️ 残高再取得エラーのためフォールバック使用: {fallback:,.0f}円" 

156 ) 

157 

158 return balance 

159 

160 except Exception as e: 

161 self.logger.error(f"❌ 残高確認エラー: {e}") 

162 raise

164 async def _run_continuous_trading(self): 

165 """連続取引実行""" 

166 try: 

167 while True: 

168 # 取引サイクル実行 

169 await self._execute_trading_cycle() 

170 self.cycle_count += 1 

171 

172 # 50サイクルごとに統計出力（約2.5時間） 

173 if self.cycle_count % 50 == 0: 

174 await self._log_progress_statistics() 

175 

176 # ライブトレード実行間隔（外部化・収益性重視） 

177 interval = get_threshold("execution.live_mode_interval_seconds", 180) 

178 await asyncio.sleep(interval) 

179 

180 except Exception as e: 

181 self.logger.error(f"❌ 連続取引実行エラー: {e}") 

182 raise

189 # 取引実行数更新（execution_service から取得）

191 if current_trades > self.trade_count: 

192 self.trade_count = current_trades 

193 self.logger.info(f"💼 ライブ取引実行: 累計{self.trade_count}件") 

194 

195 except Exception as e:
