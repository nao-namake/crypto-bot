111 def calculate_kelly_fraction(self, win_rate: float, avg_win: float, avg_loss: float) -> float: 

112 """ 

113 Kelly公式による最適ポジション比率計算 

114 

115 Args: 

116 win_rate: 勝率（0.0-1.0） 

117 avg_win: 平均利益 

118 avg_loss: 平均損失（正値で入力） 

119 

120 Returns: 

121 Kelly比率（0.0-1.0、負値の場合は0.0を返す）. 

122 """ 

123 try: 

124 if avg_loss <= 0 or win_rate <= 0 or win_rate >= 1: 

125 self.logger.warning(f"無効なパラメータ: win_rate={win_rate}, avg_loss={avg_loss}") 

126 return 0.0 

127 

128 # Kelly公式: f = (bp - q) / b 

129 # b = 平均利益/平均損失, p = 勝率, q = 敗率 

130 b = avg_win / avg_loss 

131 p = win_rate 

132 q = 1 - win_rate 

133 

134 kelly_f = (b * p - q) / b 

135 

136 # 負のKelly値は0にクリップ（取引しない） 

137 kelly_f = max(0.0, kelly_f) 

138 

139 # 理論上100%を超える場合は100%にクリップ 

140 kelly_f = min(1.0, kelly_f) 

141 

142 self.logger.debug(f"Kelly計算: b={b:.3f}, p={p:.3f}, q={q:.3f}, f={kelly_f:.3f}") 

143 

144 return kelly_f 

145 

146 except Exception as e: 

147 self.logger.error(f"Kelly計算エラー: {e}") 

148 return 0.0

150 def calculate_from_history( 

151 self, lookback_days: int = 30, strategy_filter: Optional[str] = None 

152 ) -> Optional[KellyCalculationResult]: 

153 """ 

154 取引履歴からKelly値を計算 

155 

156 Args: 

157 lookback_days: 遡る日数 

158 strategy_filter: 特定戦略のみでフィルタ（Noneで全戦略） 

159 

160 Returns: 

161 Kelly計算結果（データ不足の場合はNone）. 

162 """ 

163 try: 

164 # 期間フィルタ 

165 cutoff_date = datetime.now() - timedelta(days=lookback_days) 

166 filtered_trades = [ 

167 trade for trade in self.trade_history if trade.timestamp >= cutoff_date 

168 ] 

169 

170 # 戦略フィルタ 

171 if strategy_filter: 

172 filtered_trades = [ 

173 trade for trade in filtered_trades if trade.strategy == strategy_filter 

174 ] 

175 

176 # 最小取引数チェック 

177 if len(filtered_trades) < self.min_trades_for_kelly: 

178 self.logger.warning( 

179 f"Kelly計算に必要な取引数不足: {len(filtered_trades)} < {self.min_trades_for_kelly}" 

180 ) 

181 return None 

182 

183 # 統計計算 

184 wins = [trade.profit_loss for trade in filtered_trades if trade.is_win] 

185 losses = [abs(trade.profit_loss) for trade in filtered_trades if not trade.is_win] 

186 

187 if not wins or not losses: 

188 self.logger.warning("勝ち取引または負け取引がありません") 

189 return None 

190 

191 win_rate = len(wins) / len(filtered_trades) 

192 avg_win = sum(wins) / len(wins) 

193 avg_loss = sum(losses) / len(losses) 

194 

195 # Kelly値計算 

196 kelly_fraction = self.calculate_kelly_fraction(win_rate, avg_win, avg_loss) 

197 

198 # 安全係数適用 

199 safety_adjusted = kelly_fraction * self.safety_factor 

200 

201 # 最大ポジション制限適用 

202 recommended_size = min(safety_adjusted, self.max_position_ratio) 

203 

204 # 信頼度計算（サンプルサイズベース） 

205 confidence_level = min(1.0, len(filtered_trades) / (self.min_trades_for_kelly * 2)) 

206 

207 result = KellyCalculationResult( 

208 kelly_fraction=kelly_fraction, 

209 win_rate=win_rate, 

210 avg_win_loss_ratio=avg_win / avg_loss, 

211 safety_adjusted_fraction=safety_adjusted, 

212 recommended_position_size=recommended_size, 

213 sample_size=len(filtered_trades), 

214 confidence_level=confidence_level, 

215 ) 

216 

217 self.logger.info( 

218 f"Kelly計算完了: Kelly={kelly_fraction:.3f}, " 

219 f"調整後={safety_adjusted:.3f}, 推奨={recommended_size:.3f}, " 

220 f"勝率={win_rate:.1%}, サンプル={len(filtered_trades)}" 

221 ) 

222 

223 return result 

224 

225 except Exception as e: 

226 self.logger.error(f"履歴からのKelly計算エラー: {e}") 

227 return None

229 def calculate_optimal_size( 

230 self, 

231 ml_confidence: float, 

232 strategy_name: str = "default", 

233 expected_return: Optional[float] = None, 

234 ) -> float: 

235 """ 

236 ML予測信頼度を考慮した最適ポジションサイズ計算 

237 

238 Args: 

239 ml_confidence: ML予測の信頼度（0.0-1.0） 

240 strategy_name: 戦略名 

241 expected_return: 期待リターン（Noneの場合は履歴ベース） 

242 

243 Returns: 

244 推奨ポジションサイズ（最大3%制限済み）. 

245 """ 

246 try: 

247 # 履歴ベースのKelly値取得 

248 kelly_result = self.calculate_from_history(strategy_filter=strategy_name) 

249 

250 if kelly_result is None: 

251 # 履歴データ不足の場合は保守的なサイズ 

252 conservative_size = 0.01 * ml_confidence # 1% * 信頼度 

253 self.logger.warning(f"Kelly履歴不足、保守的サイズ使用: {conservative_size:.3f}") 

254 return min(conservative_size, self.max_position_ratio) 

255 

256 # ML信頼度による調整 

257 confidence_adjusted_size = kelly_result.recommended_position_size * ml_confidence 

258 

259 # データ信頼度による調整 

260 data_confidence_adjusted = confidence_adjusted_size * kelly_result.confidence_level 

261 

262 # 最終制限適用 

263 final_size = min(data_confidence_adjusted, self.max_position_ratio) 

264 

265 self.logger.debug( 

266 f"最適サイズ計算: Kelly推奨={kelly_result.recommended_position_size:.3f}, " 

267 f"ML信頼度={ml_confidence:.3f}, 最終={final_size:.3f}" 

268 ) 

269 

270 return final_size 

271 

272 except Exception as e: 

273 self.logger.error(f"最適サイズ計算エラー: {e}") 

274 return 0.01 # フォールバック値

276 def calculate_dynamic_position_size( 

277 self, 

278 balance: float, 

279 entry_price: float, 

280 atr_value: float, 

281 ml_confidence: float, 

282 target_volatility: float = 0.01, 

283 max_scale: float = 3.0, 

284 ) -> Tuple[float, float]: 

285 """ 

286 ボラティリティ連動ダイナミックポジションサイジング 

287 

288 レガシーシステムの優秀な機能を統合。ボラティリティに応じて 

289 ポジションサイズを調整し、高ボラ時は小さく、低ボラ時は大きくする。 

290 

291 Args: 

292 balance: 現在の口座残高 

293 entry_price: エントリー予定価格 

294 atr_value: 現在のATR値 

295 ml_confidence: ML予測信頼度 

296 target_volatility: 目標ボラティリティ（0.01 = 1%） 

297 max_scale: 最大スケール倍率 

298 

299 Returns: 

300 (調整済みポジションサイズ, ストップロス価格). 

301 """ 

302 try: 

303 # 入力検証 

304 if balance <= 0: 

305 raise ValueError(f"残高は正値である必要があります: {balance}") 

306 if entry_price <= 0: 

307 raise ValueError(f"エントリー価格は正値である必要があります: {entry_price}") 

308 if atr_value < 0: 

309 raise ValueError(f"ATR値は非負値である必要があります: {atr_value}") 

310 if not 0 < target_volatility <= 1.0: 

311 raise ValueError(f"目標ボラティリティは0-1.0の範囲: {target_volatility}") 

312 

313 # 1) ベースKellyサイズ計算 

314 base_kelly_size = self.calculate_optimal_size( 

315 ml_confidence=ml_confidence, strategy_name="dynamic" 

316 ) 

317 

318 # 2) ATRベースのストップロス計算（レガシー手法を活用） 

319 stop_atr_multiplier = 2.0 # デフォルト 

320 stop_loss_price = entry_price - (atr_value * stop_atr_multiplier) 

321 

322 # ストップロス安全チェック 

323 if stop_loss_price <= 0: 

324 self.logger.warning( 

325 f"計算されたストップロス価格が負値: {stop_loss_price:.2f}, " 

326 f"エントリー価格の1%に設定" 

327 ) 

328 stop_loss_price = entry_price * 0.99 

329 

330 # 3) ボラティリティスケール計算 

331 if atr_value == 0: 

332 volatility_pct = target_volatility # フォールバック 

333 self.logger.warning("ATR値が0、目標ボラティリティを使用") 

334 else: 

335 volatility_pct = atr_value / entry_price 

336 

337 # スケール係数（ボラティリティが高いほど小さくなる） 

338 if volatility_pct <= 0: 

339 scale = 1.0 

340 else: 

341 scale = target_volatility / volatility_pct 

342 

343 # スケール制限適用 

344 scale = max(0.1, min(scale, max_scale)) 

345 

346 # 4) 最終ポジションサイズ計算 

347 dynamic_position_size = base_kelly_size * scale 

348 

349 # 5) 安全制限適用（レガシーの安全機能を活用） 

350 # 口座の30%以内に制限 

351 max_safe_position = min( 

352 balance * 0.3 / entry_price, # 口座の30% 

353 balance * self.max_position_ratio, # Kelly最大制限 

354 ) 

355 

356 final_position_size = min(dynamic_position_size, max_safe_position) 

357 

358 self.logger.info( 

359 f"ダイナミックポジションサイジング: " 

360 f"ベースKelly={base_kelly_size:.4f}, スケール={scale:.2f}, " 

361 f"最終サイズ={final_position_size:.4f}, ATR={atr_value:.6f}" 

362 ) 

363 

364 return final_position_size, stop_loss_price 

365 

366 except Exception as e: 

367 self.logger.error(f"ダイナミックポジションサイジングエラー: {e}") 

368 # 複数レベルフォールバック（レガシー手法活用） 

369 return self._safe_fallback_position_size(balance, entry_price)

371 def _safe_fallback_position_size( 

372 self, balance: float, entry_price: float 

373 ) -> Tuple[float, float]: 

374 """ 

375 安全なフォールバックポジションサイズ計算 

376 

377 レガシーシステムの複数レベルフォールバック機能を統合. 

378 """ 

379 try: 

380 # レベル1: 最小安全サイズ 

381 safe_position = balance * 0.01 / entry_price # 口座の1% 

382 safe_stop = entry_price * 0.95 # 5%下のストップ 

383 

384 # レベル2: 最大制限チェック 

385 max_safe = balance * 0.1 / entry_price # 最大10% 

386 final_position = min(safe_position, max_safe) 

387 

388 self.logger.warning( 

389 f"フォールバックポジションサイズ使用: {final_position:.4f}, " 

390 f"ストップ: {safe_stop:.2f}" 

391 ) 

392 

393 return final_position, safe_stop 

394 

395 except Exception as fallback_error: 

396 self.logger.critical(f"フォールバック計算も失敗: {fallback_error}") 

397 # 最終安全値 

398 emergency_position = balance * 0.005 / entry_price # 0.5% 

399 emergency_stop = entry_price * 0.98 # 2%下 

400 return emergency_position, emergency_stop
