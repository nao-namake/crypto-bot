16class MarketUncertaintyCalculator: 

17 """ 

18 市場不確実性計算統合クラス 

19 

20 全戦略で使用する市場不確実性（market uncertainty）計算を一元化。 

21 ATRベースのボラティリティ・ボリューム異常度・価格変動率を統合し、 

22 市場状況を反映した動的信頼度調整係数を提供する。 

23 

24 Phase 38.4: 全戦略から重複コード削除・統一ロジック実装 

25 """ 

26 

27 _logger = None 

28 

29 @classmethod 

30 def _get_logger(cls): 

31 """ロガーの遅延初期化（循環インポート回避）""" 

32 if cls._logger is None: 

33 cls._logger = get_logger() 

34 return cls._logger

36 @classmethod 

37 def calculate(cls, df: pd.DataFrame) -> float: 

38 """ 

39 市場データ基づく不確実性計算（設定ベース統一ロジック） 

40 

41 全戦略共通の市場不確実性計算を実行。 

42 thresholds.yamlの設定値を使用し、市場状況を0-0.1の範囲で数値化する。 

43 

44 Args: 

45 df: 市場データ（close, atr_14, volume列必須） 

46 

47 Returns: 

48 float: 市場不確実性係数（0-0.1の範囲、設定値で制限） 

49 

50 Raises: 

51 なし（エラー時はデフォルト値0.02を返す） 

52 

53 Example: 

54 >>> uncertainty = MarketUncertaintyCalculator.calculate(df) 

55 >>> adjusted_confidence = base_confidence * (1 + uncertainty) 

56 """ 

57 logger = cls._get_logger() 

58 

59 try: 

60 # 循環インポート回避のため遅延インポート 

61 from ...core.config.threshold_manager import get_threshold 

62 

63 # 設定値取得（thresholds.yaml: dynamic_confidence.market_uncertainty） 

64 volatility_max = get_threshold( 

65 "dynamic_confidence.market_uncertainty.volatility_factor_max", 0.05 

66 ) 

67 volume_max = get_threshold( 

68 "dynamic_confidence.market_uncertainty.volume_factor_max", 0.03 

69 ) 

70 volume_multiplier = get_threshold( 

71 "dynamic_confidence.market_uncertainty.volume_multiplier", 0.1 

72 ) 

73 price_max = get_threshold( 

74 "dynamic_confidence.market_uncertainty.price_factor_max", 0.02 

75 ) 

76 uncertainty_max = get_threshold( 

77 "dynamic_confidence.market_uncertainty.uncertainty_max", 0.10 

78 ) 

79 

80 # 1. ATRベースのボラティリティ要因 

81 current_price = float(df["close"].iloc[-1]) 

82 atr_value = float(df["atr_14"].iloc[-1]) 

83 volatility_factor = min(volatility_max, atr_value / current_price) 

84 

85 # 2. ボリューム異常度（平均からの乖離） 

86 current_volume = float(df["volume"].iloc[-1]) 

87 avg_volume = float(df["volume"].rolling(20).mean().iloc[-1]) 

88 volume_ratio = current_volume / avg_volume if avg_volume > 0 else 1.0 

89 volume_factor = min(volume_max, abs(volume_ratio - 1.0) * volume_multiplier) 

90 

91 # 3. 価格変動率（短期動向） 

92 price_change = abs(float(df["close"].pct_change().iloc[-1])) 

93 price_factor = min(price_max, price_change) 

94 

95 # 統合不確実性（設定値の範囲で市場状況を反映） 

96 market_uncertainty = volatility_factor + volume_factor + price_factor 

97 

98 # 設定値で調整範囲を制限 

99 result = min(uncertainty_max, market_uncertainty) 

100 

101 logger.debug( 

102 f"[MarketUncertainty] 計算完了: {result:.4f} " 

103 f"(volatility={volatility_factor:.4f}, volume={volume_factor:.4f}, price={price_factor:.4f})" 

104 ) 

105 

106 return result 

107 

108 except Exception as e: 

109 logger.warning(f"[MarketUncertainty] 計算エラー: {e}") 

110 return 0.02 # デフォルト値（2%の軽微な調整）
