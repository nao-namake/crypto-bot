38 """ 

39 全テクニカル指標を生成

89 def _calculate_rsi(self, close: pd.Series, period: int = 14) -> pd.Series: 

90 """RSI計算.""" 

91 delta = close.diff() 

92 gain = (delta.where(delta > 0, 0)).rolling(window=period, min_periods=1).mean() 

93 loss = (-delta.where(delta < 0, 0)).rolling(window=period, min_periods=1).mean() 

94 rs = gain / (loss + 1e-8) 

95 return 100 - (100 / (1 + rs)) 

96 

97 def _calculate_macd(self, close: pd.Series) -> tuple: 

98 """MACD計算（MACDラインとシグナルラインを返す）.""" 

99 exp1 = close.ewm(span=12, adjust=False).mean() 

100 exp2 = close.ewm(span=26, adjust=False).mean() 

101 macd_line = exp1 - exp2 

102 macd_signal = macd_line.ewm(span=9, adjust=False).mean() 

103 return macd_line, macd_signal 

104 

105 def _calculate_atr(self, df: pd.DataFrame, period: int = 14) -> pd.Series: 

106 """ATR計算.""" 

107 high_low = df["high"] - df["low"] 

108 high_close = np.abs(df["high"] - df["close"].shift()) 

109 low_close = np.abs(df["low"] - df["close"].shift()) 

110 true_range = np.maximum(high_low, np.maximum(high_close, low_close)) 

111 return true_range.rolling(window=period, min_periods=1).mean() 

112 

113 def _calculate_bb_position(self, close: pd.Series, period: int = 20) -> pd.Series: 

114 """ボリンジャーバンド位置計算.""" 

115 bb_middle = close.rolling(window=period, min_periods=1).mean() 

116 bb_std_dev = close.rolling(window=period, min_periods=1).std() 

117 bb_upper = bb_middle + (bb_std_dev * 2) 

118 bb_lower = bb_middle - (bb_std_dev * 2) 

119 return (close - bb_lower) / (bb_upper - bb_lower + 1e-8)
