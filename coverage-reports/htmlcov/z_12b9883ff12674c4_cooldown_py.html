28 async def should_apply_cooldown(self, evaluation: TradeEvaluation) -> bool: 

29 """ 

30 Phase 31.1: 柔軟なクールダウン判定 

31 

32 強いトレンド発生時はクールダウンをスキップし、 

33 機会損失を防ぐ。 

34 

35 Args: 

36 evaluation: 取引評価結果（market_conditionsを含む） 

37 

38 Returns: 

39 bool: クールダウンを適用するか 

40 """ 

41 try: 

42 # features.yaml から設定取得 

43 features = get_features_config() 

44 features_config = features.get("trading", {}).get("cooldown", {}) 

45 

46 # クールダウン無効の場合は適用しない 

47 if not features_config.get("enabled", True): 

48 return False 

49 

50 # 柔軟モード無効の場合は常に適用 

51 if not features_config.get("flexible_mode", False): 

52 return True 

53 

54 # 柔軟モード: トレンド強度を判定 

55 market_data = evaluation.market_conditions.get("market_data") 

56 if market_data is None: 

57 # 市場データがない場合はデフォルトで適用 

58 return True 

59 

60 trend_strength = self.calculate_trend_strength(market_data) 

61 threshold = features_config.get("trend_strength_threshold", 0.7) 

62 

63 # 強いトレンド時はクールダウンをスキップ 

64 if trend_strength >= threshold: 

65 self.logger.info( 

66 f"🔥 強トレンド検出 (強度: {trend_strength:.2f}) - クールダウンスキップ" 

67 ) 

68 return False 

69 

70 return True 

71 

72 except Exception as e: 

73 self.logger.warning(f"⚠️ クールダウン判定エラー: {e} - デフォルトで適用") 

74 return True

76 def calculate_trend_strength(self, market_data: Dict) -> float: 

77 """ 

78 Phase 31.1: トレンド強度計算（ADX・DI・EMA総合判定） 

79 

80 Args: 

81 market_data: 市場データ（特徴量含む） 

82 

83 Returns: 

84 float: トレンド強度 (0.0-1.0) 

85 """ 

86 try: 

87 # 4h足データを使用してトレンド強度を判定 

88 df = market_data.get("4h", pd.DataFrame()) 

89 if df.empty or len(df) < 3: 

90 return 0.0 

91 

92 # ADX（トレンド強度指標） 

93 adx = float(df["adx_14"].iloc[-1]) if "adx_14" in df.columns else 0.0 

94 

95 # DI差分（方向性） 

96 plus_di = float(df["plus_di_14"].iloc[-1]) if "plus_di_14" in df.columns else 0.0 

97 minus_di = float(df["minus_di_14"].iloc[-1]) if "minus_di_14" in df.columns else 0.0 

98 di_diff = abs(plus_di - minus_di) 

99 

100 # EMAトレンド（方向の一貫性） 

101 ema_20 = float(df["ema_20"].iloc[-1]) if "ema_20" in df.columns else 0.0 

102 ema_50 = float(df["ema_50"].iloc[-1]) if "ema_50" in df.columns else 0.0 

103 ema_trend = abs(ema_20 - ema_50) / ema_50 if ema_50 > 0 else 0.0 

104 

105 # トレンド強度スコア算出 

106 # ADX: 25以上で強いトレンド（正規化: 0-50 → 0-1） 

107 adx_score = min(1.0, adx / 50.0) 

108 

109 # DI差分: 20以上で明確な方向性（正規化: 0-40 → 0-1） 

110 di_score = min(1.0, di_diff / 40.0) 

111 

112 # EMAトレンド: 2%以上で明確なトレンド（正規化: 0-5% → 0-1） 

113 ema_score = min(1.0, ema_trend / 0.05) 

114 

115 # 加重平均（ADX重視: 50%、DI: 30%、EMA: 20%） 

116 trend_strength = adx_score * 0.5 + di_score * 0.3 + ema_score * 0.2 

117 

118 self.logger.debug( 

119 f"トレンド強度計算: ADX={adx:.1f}({adx_score:.2f}), " 

120 f"DI差={di_diff:.1f}({di_score:.2f}), " 

121 f"EMAトレンド={ema_trend:.3f}({ema_score:.2f}) → 総合={trend_strength:.2f}" 

122 ) 

123 

124 return trend_strength 

125 

126 except Exception as e: 

127 self.logger.error(f"❌ トレンド強度計算エラー: {e}") 

128 return 0.0

130 def get_cooldown_status(self, trend_strength: float) -> Dict[str, Any]: 

131 """ 

132 クールダウンステータスを取得 

133 

134 Args: 

135 trend_strength: トレンド強度 

136 

137 Returns: 

138 クールダウン状態情報 

139 """ 

140 try: 

141 features = get_features_config() 

142 features_config = features.get("trading", {}).get("cooldown", {}) 

143 

144 enabled = features_config.get("enabled", True) 

145 flexible_mode = features_config.get("flexible_mode", False) 

146 threshold = features_config.get("trend_strength_threshold", 0.7) 

147 

148 status = { 

149 "enabled": enabled, 

150 "flexible_mode": flexible_mode, 

151 "trend_strength": trend_strength, 

152 "threshold": threshold, 

153 "skip_cooldown": False, 

154 "reason": "", 

155 } 

156 

157 if not enabled: 

158 status["reason"] = "クールダウン無効" 

159 elif not flexible_mode: 

160 status["reason"] = "通常モード（常に適用）" 

161 elif trend_strength >= threshold: 

162 status["skip_cooldown"] = True 

163 status["reason"] = f"強トレンド検出（{trend_strength:.2f} >= {threshold}）" 

164 else: 

165 status["reason"] = f"トレンド弱（{trend_strength:.2f} < {threshold}）" 

166 

167 return status 

168 

169 except Exception as e: 

170 self.logger.error(f"❌ クールダウンステータス取得エラー: {e}") 

171 return { 

172 "enabled": True, 

173 "flexible_mode": False, 

174 "trend_strength": 0.0, 

175 "threshold": 0.7, 

176 "skip_cooldown": False, 

177 "reason": f"エラー: {e}", 

178 }
