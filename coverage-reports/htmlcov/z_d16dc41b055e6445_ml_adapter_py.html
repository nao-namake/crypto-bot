1""" 

2MLサービス統合アダプター - Phase 22 分割最適化 

3 

4ProductionEnsembleとEnsembleModelの統一インターフェースを提供し、 

5MLモデル未学習エラーを根本的に解決するアダプター。 

6 

7Phase 22での分割により、責任を明確化： 

8- ml_loader.py: モデル読み込み専門 

9- ml_fallback.py: フォールバック機能専門 

10- ml_adapter.py: 統合インターフェース・予測機能 

11"""

14 

15import numpy as np 

16import pandas as pd 

17 

18from ..config import get_threshold 

19from ..exceptions import ModelPredictionError 

20from ..logger import CryptoBotLogger 

21from .ml_fallback import DummyModel 

22from .ml_loader import MLModelLoader

25class MLServiceAdapter: 

26 """ 

27 MLサービス統合アダプター - Phase 22 最適化版 

28 

29 MLModelLoaderに読み込み処理を委譲し、予測インターフェースを提供。 

30 モジュラー設計により保守性と可読性を大幅向上。 

31 """ 

32 

33 def __init__(self, logger: CryptoBotLogger) -> None: 

34 """ 

35 MLServiceAdapter初期化 

36 

37 Args: 

38 logger: ログシステム 

39 """ 

40 self.logger = logger 

41 self.loader = MLModelLoader(logger) 

42 

43 # モデル情報をローダーから取得 

44 self.model = self.loader.load_model_with_priority() 

45 self.model_type = self.loader.model_type 

46 self.is_fitted = self.loader.is_fitted 

47 

48 def predict( 

49 self, X: Union[pd.DataFrame, np.ndarray], use_confidence: bool = True 

50 ) -> np.ndarray: 

51 """ 

52 統一predict インターフェース 

53 

54 Args: 

55 X: 特徴量データ 

56 use_confidence: 信頼度閾値使用（EnsembleModelのみ） 

57 

58 Returns: 

59 np.ndarray: 予測結果 

60 """ 

61 if not self.is_fitted: 

62 raise ValueError("モデルが学習されていません") 

63 

64 try: 

65 # EnsembleModelの場合は use_confidence パラメータを渡す 

66 if ( 

67 hasattr(self.model, "predict") 

68 and "use_confidence" in self.model.predict.__code__.co_varnames 

69 ): 

70 return self.model.predict(X, use_confidence=use_confidence) 

71 else: 

72 return self.model.predict(X) 

73 

74 except Exception as e: 

75 self.logger.error(f"予測エラー: {e}") 

76 # エラー時はダミーモデルにフォールバック 

77 if self.model_type != "DummyModel": 

78 self.logger.warning("エラーによりダミーモデルにフォールバック") 

79 dummy_model = DummyModel() 

80 return dummy_model.predict(X) 

81 else: 

82 raise ModelPredictionError(f"ダミーモデルでも予測に失敗: {e}") 

83 

84 def predict_proba(self, X: Union[pd.DataFrame, np.ndarray]) -> np.ndarray: 

85 """ 

86 統一predict_proba インターフェース 

87 

88 Args: 

89 X: 特徴量データ 

90 

91 Returns: 

92 np.ndarray: 予測確率 

93 """ 

94 if not self.is_fitted: 

95 raise ValueError("モデルが学習されていません") 

96 

97 try: 

98 return self.model.predict_proba(X) 

99 except Exception as e: 

100 self.logger.error(f"確率予測エラー: {e}") 

101 if self.model_type != "DummyModel": 

102 self.logger.warning("確率予測エラーによりダミーモデルにフォールバック") 

103 dummy_model = DummyModel() 

104 return dummy_model.predict_proba(X) 

105 else: 

106 raise ModelPredictionError(f"ダミーモデルでも確率予測に失敗: {e}")

108 def get_model_info(self) -> Dict[str, Any]: 

109 """モデル情報取得 - ローダーから情報を取得""" 

110 base_info = { 

111 "adapter_type": "MLServiceAdapter", 

112 "adapter_version": "Phase22_Optimized",

114 

115 # ローダーからモデル情報を取得 

116 loader_info = self.loader.get_model_info() 

117 base_info.update(loader_info)

121 def reload_model(self) -> bool: 

122 """モデル再読み込み（復旧用）""" 

123 self.logger.info("🔄 MLモデル再読み込み開始") 

124 

125 try: 

126 # ローダーを使ってモデル再読み込み 

127 if self.loader.reload_model(): 

128 # 成功した場合、アダプターの状態を更新 

129 self.model = self.loader.model 

130 self.model_type = self.loader.model_type 

131 self.is_fitted = self.loader.is_fitted 

132 return True 

133 else: 

134 return False 

135 

136 except Exception as e: 

137 self.logger.error(f"MLモデル再読み込み予期しないエラー: {e}") 

138 return False
