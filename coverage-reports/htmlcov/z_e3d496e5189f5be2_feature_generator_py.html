221 def _calculate_rsi(self, close: pd.Series, period: int = 14) -> pd.Series: 

222 """RSI計算""" 

223 delta = close.diff() 

224 gain = (delta.where(delta > 0, 0)).rolling(window=period, min_periods=1).mean() 

225 loss = (-delta.where(delta < 0, 0)).rolling(window=period, min_periods=1).mean() 

226 rs = gain / (loss + 1e-8) 

227 return 100 - (100 / (1 + rs)) 

228 

229 def _calculate_macd(self, close: pd.Series) -> tuple: 

230 """MACD計算（MACDラインとシグナルラインを返す）""" 

231 exp1 = close.ewm(span=12, adjust=False).mean() 

232 exp2 = close.ewm(span=26, adjust=False).mean() 

233 macd_line = exp1 - exp2 

234 macd_signal = macd_line.ewm(span=9, adjust=False).mean() 

235 return macd_line, macd_signal 

236 

237 def _calculate_atr(self, df: pd.DataFrame, period: int = 14) -> pd.Series: 

238 """ATR計算""" 

239 high_low = df["high"] - df["low"] 

240 high_close = np.abs(df["high"] - df["close"].shift()) 

241 low_close = np.abs(df["low"] - df["close"].shift()) 

242 true_range = np.maximum(high_low, np.maximum(high_close, low_close)) 

243 return true_range.rolling(window=period, min_periods=1).mean() 

244 

245 def _calculate_bb_position(self, close: pd.Series, period: int = 20) -> pd.Series: 

246 """ボリンジャーバンド位置計算""" 

247 bb_middle = close.rolling(window=period, min_periods=1).mean() 

248 bb_std_dev = close.rolling(window=period, min_periods=1).std() 

249 bb_upper = bb_middle + (bb_std_dev * 2) 

250 bb_lower = bb_middle - (bb_std_dev * 2) 

251 return (close - bb_lower) / (bb_upper - bb_lower + 1e-8) 

252 

253 def _calculate_zscore(self, close: pd.Series, period: Optional[int] = None) -> pd.Series: 

254 """Z-Score計算""" 

255 try: 

256 if period is None: 

257 period = get_anomaly_config("zscore.calculation_period", 20) 

258 rolling_mean = close.rolling(window=period, min_periods=1).mean() 

259 rolling_std = close.rolling(window=period, min_periods=1).std() 

260 return (close - rolling_mean) / (rolling_std + 1e-8) 

261 except Exception as e: 

262 self.logger.error(f"Z-Score計算エラー: {e}") 

263 return pd.Series(np.zeros(len(close)), index=close.index)

265 def _calculate_volume_ratio(self, volume: pd.Series, period: Optional[int] = None) -> pd.Series: 

266 """出来高比率計算""" 

267 try: 

268 if period is None: 

269 period = get_anomaly_config("volume_ratio.calculation_period", 20) 

270 volume_avg = volume.rolling(window=period, min_periods=1).mean() 

271 return volume / (volume_avg + 1e-8) 

272 except Exception as e: 

273 self.logger.error(f"出来高比率計算エラー: {e}") 

274 return pd.Series(np.zeros(len(volume)), index=volume.index)

276 def _calculate_market_stress(self, df: pd.DataFrame) -> pd.Series: 

277 """市場ストレス度指標計算（統合異常指標）""" 

278 try: 

279 # 価格ギャップ（前日比で大きな価格変動） 

280 price_gap = np.abs(df["open"] - df["close"].shift(1)) / df["close"].shift(1) 

281 

282 # 日中変動率（High-Low range） 

283 intraday_range = (df["high"] - df["low"]) / df["close"] 

284 

285 # 出来高スパイク（平均の何倍か） 

286 volume_avg = df["volume"].rolling(window=self.lookback_period, min_periods=1).mean() 

287 volume_spike = df["volume"] / (volume_avg + 1e-8) 

288 

289 # 重み付け合成（設定ファイルから取得） 

290 price_weight = get_anomaly_config("market_stress.price_gap_weight", 0.4) 

291 intraday_weight = get_anomaly_config("market_stress.intraday_range_weight", 0.3) 

292 volume_weight = get_anomaly_config("market_stress.volume_spike_weight", 0.3) 

293 

294 market_stress = ( 

295 price_weight * self._normalize(price_gap) 

296 + intraday_weight * self._normalize(intraday_range) 

297 + volume_weight * self._normalize(volume_spike) 

298 ) 

299 

300 return market_stress 

301 

302 except Exception as e: 

303 self.logger.error(f"市場ストレス度指標エラー: {e}") 

304 return pd.Series(np.zeros(len(df)), index=df.index) 

305 

306 def _normalize(self, series: pd.Series) -> pd.Series: 

307 """0-1範囲に正規化""" 

308 try: 

309 # 外れ値処理（設定ファイルから取得） 

310 outlier_clip_quantile = get_anomaly_config("normalization.outlier_clip_quantile", 0.95) 

311 upper_bound = series.quantile(outlier_clip_quantile) 

312 clipped_series = np.clip(series, 0, upper_bound) 

313 

314 # 0-1正規化 

315 min_val = clipped_series.min() 

316 max_val = clipped_series.max() 

317 

318 if max_val - min_val == 0: 

319 return pd.Series(np.zeros(len(series)), index=series.index) 

320 

321 return (clipped_series - min_val) / (max_val - min_val) 

322 

323 except Exception: 

324 return pd.Series(np.zeros(len(series)), index=series.index)

370 def generate_features_sync(self, df: pd.DataFrame) -> pd.DataFrame: 

371 """ 

372 同期特徴量生成（後方互換性用） 

373 

374 Phase 18統合により、テストとの互換性のため同期版を提供。 

375 DataFrame入力・DataFrame出力で、異常検知特徴量のみ生成。 

376 

377 Args: 

378 df: OHLCVデータフレーム 

379 Returns: 

380 特徴量追加されたデータフレーム 

381 """ 

382 # 必要列のチェック（OHLCV全て必要） 

383 required_columns = ["open", "high", "low", "close", "volume"] 

384 missing_columns = [col for col in required_columns if col not in df.columns] 

385 if missing_columns: 

386 from src.core.exceptions import DataProcessingError 

387 

388 raise DataProcessingError(f"必要列が不足: {missing_columns}") 

389 

390 # 簡易版：market_stress特徴量のみ生成 

391 result_df = df.copy() 

392 

393 # 効果的なlookback_periodを決定（最小3、最大はデータ長の80%） 

394 effective_lookback = min(self.lookback_period, max(3, int(len(df) * 0.8))) 

395 

396 # 出来高異常検知 (volume_ratio) 

397 if "volume" in df.columns: 

398 if len(df) >= 2: 

399 rolling_mean = df["volume"].rolling(window=effective_lookback, min_periods=1).mean() 

400 volume_ratio = df["volume"] / rolling_mean.fillna(df["volume"].mean()) 

401 result_df["volume_ratio"] = volume_ratio.fillna(1.0) 

402 else: 

403 result_df["volume_ratio"] = 1.0 

404 

405 # 価格Z-Score (zscore) 

406 if "close" in df.columns: 

407 if len(df) >= 2: 

408 rolling_mean = df["close"].rolling(window=effective_lookback, min_periods=1).mean() 

409 rolling_std = df["close"].rolling(window=effective_lookback, min_periods=1).std() 

410 zscore = (df["close"] - rolling_mean) / rolling_std.fillna( 

411 1.0 

412 ) # std=0の場合は1.0で除算 

413 result_df["zscore"] = zscore.fillna(0.0) 

414 else: 

415 result_df["zscore"] = 0.0 

416 

417 # 市場ストレス (market_stress) - 統合指標 

418 stress_values = [] 

419 for i in range(len(df)): 

420 volume_stress = ( 

421 result_df.loc[i, "volume_ratio"] if "volume_ratio" in result_df.columns else 1.0 

422 ) 

423 price_stress = abs(result_df.loc[i, "zscore"]) if "zscore" in result_df.columns else 0.0 

424 

425 # 正規化 (0-1) - より敏感に調整 

426 volume_stress = min(volume_stress / 3.0, 1.0) # 3倍以上で1.0に変更 

427 price_stress = min(price_stress / 2.0, 1.0) # 2σ以上で1.0に変更 

428 

429 market_stress = (volume_stress + price_stress) / 2.0 

430 stress_values.append(market_stress) 

431 

432 result_df["market_stress"] = stress_values 

433 

434 # computed_featuresを更新 

435 self.computed_features.update(["volume_ratio", "zscore", "market_stress"]) 

436 

437 return result_df
