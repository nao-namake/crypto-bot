41 """ 

42 取引サイクル実行 

43 

44 高レベル取引サイクル制御。各Phase層に具体的な処理を委譲し、 

45 ここでは統合フローの制御のみ実行。 

46 """ 

47 cycle_id = datetime.now().isoformat() 

48 self.logger.info(f"取引サイクル開始 - ID: {cycle_id}") 

49 

50 try:

52 market_data = await self._fetch_market_data() 

53 if market_data is None: 

54 self.logger.warning("市場データ取得失敗 - サイクル終了") 

55 return

58 features, main_features = await self._generate_features(market_data) 

59 

60 # Phase 4: 戦略評価（Phase 31: マルチタイムフレーム対応） 

61 strategy_signal = await self._evaluate_strategy(main_features, features)

76 await self._check_stop_conditions(cycle_id)

78 except ValueError as e:

80 except ModelLoadError as e: 

81 await self._handle_model_error(e, cycle_id) 

82 except (ConnectionError, TimeoutError) as e:

84 except (AttributeError, TypeError) as e:

348 def _integrate_ml_with_strategy( 

349 self, ml_prediction: dict, strategy_signal: StrategySignal 

350 ) -> StrategySignal: 

351 """ 

352 Phase 29.5: ML予測と戦略シグナルの統合 

353 

354 ML予測結果を戦略シグナルと統合し、最終的な取引信頼度を調整。 

355 一致時はボーナス、不一致時はペナルティを適用。 

356 

357 Args: 

358 ml_prediction: ML予測結果 {"prediction": int, "confidence": float} 

359 prediction: -1=売り, 0=保持, 1=買い 

360 strategy_signal: 戦略シグナル (StrategySignalオブジェクト) 

361 

362 Returns: 

363 StrategySignal: 統合後のシグナル（ML調整済み） 

364 """ 

365 try: 

366 # ML統合が無効の場合は元のシグナルをそのまま返す 

367 if not get_threshold("ml.strategy_integration.enabled", False): 

368 self.logger.debug("ML統合無効 - 戦略シグナルをそのまま使用") 

369 return strategy_signal 

370 

371 # ML予測信頼度が低い場合は統合しない 

372 ml_confidence = ml_prediction.get("confidence", 0.0) 

373 min_ml_confidence = get_threshold("ml.strategy_integration.min_ml_confidence", 0.6) 

374 

375 if ml_confidence < min_ml_confidence: 

376 self.logger.info( 

377 f"ML信頼度不足 ({ml_confidence:.3f} < {min_ml_confidence:.3f}) - 戦略シグナルのみ使用" 

378 ) 

379 return strategy_signal 

380 

381 # 予測値とアクションの変換 

382 ml_pred = ml_prediction.get("prediction", 0) 

383 ml_action_map = {-1: "sell", 0: "hold", 1: "buy"} 

384 ml_action = ml_action_map.get(ml_pred, "hold") 

385 

386 # StrategySignalオブジェクトから属性を直接取得 

387 strategy_action = strategy_signal.action 

388 strategy_confidence = strategy_signal.confidence 

389 

390 self.logger.info( 

391 f"🔄 ML統合開始: 戦略={strategy_action}({strategy_confidence:.3f}), " 

392 f"ML={ml_action}({ml_confidence:.3f})" 

393 ) 

394 

395 # 一致・不一致判定 

396 is_agreement = (ml_action == strategy_action) or ( 

397 ml_action == "hold" and strategy_action in ["buy", "sell"] 

398 ) 

399 

400 # 統合重み取得 

401 ml_weight = get_threshold("ml.strategy_integration.ml_weight", 0.3) 

402 strategy_weight = get_threshold("ml.strategy_integration.strategy_weight", 0.7) 

403 high_confidence_threshold = get_threshold( 

404 "ml.strategy_integration.high_confidence_threshold", 0.8 

405 ) 

406 

407 # ベース信頼度計算（加重平均） 

408 base_confidence = (strategy_confidence * strategy_weight) + (ml_confidence * ml_weight) 

409 

410 # ML高信頼度時のボーナス・ペナルティ適用 

411 if ml_confidence >= high_confidence_threshold: 

412 if is_agreement: 

413 # 一致時: ボーナス適用 

414 agreement_bonus = get_threshold("ml.strategy_integration.agreement_bonus", 1.2) 

415 adjusted_confidence = min(base_confidence * agreement_bonus, 1.0) 

416 self.logger.info( 

417 f"✅ ML・戦略一致（ML高信頼度） - ボーナス適用: " 

418 f"{base_confidence:.3f} → {adjusted_confidence:.3f}" 

419 ) 

420 else: 

421 # 不一致時: ペナルティ適用 

422 disagreement_penalty = get_threshold( 

423 "ml.strategy_integration.disagreement_penalty", 0.7 

424 ) 

425 adjusted_confidence = base_confidence * disagreement_penalty 

426 self.logger.warning( 

427 f"⚠️ ML・戦略不一致（ML高信頼度） - ペナルティ適用: " 

428 f"{base_confidence:.3f} → {adjusted_confidence:.3f}" 

429 ) 

430 

431 # 不一致時はholdに変更する選択肢も 

432 if adjusted_confidence < 0.4: # 信頼度が極端に低い場合 

433 self.logger.warning( 

434 f"⛔ 信頼度極低（{adjusted_confidence:.3f}）- holdに変更" 

435 ) 

436 # 新しいStrategySignalオブジェクトとして返す（hold変更） 

437 from ...strategies.base.strategy_base import StrategySignal 

438 

439 return StrategySignal( 

440 strategy_name=strategy_signal.strategy_name, 

441 timestamp=strategy_signal.timestamp, 

442 action="hold", 

443 confidence=adjusted_confidence, 

444 strength=adjusted_confidence, 

445 current_price=strategy_signal.current_price, 

446 entry_price=strategy_signal.entry_price, 

447 stop_loss=strategy_signal.stop_loss, 

448 take_profit=strategy_signal.take_profit, 

449 position_size=strategy_signal.position_size, 

450 risk_ratio=strategy_signal.risk_ratio, 

451 indicators=strategy_signal.indicators, 

452 reason=f"ML・戦略不一致（信頼度極低）", 

453 metadata={ 

454 **(strategy_signal.metadata or {}), 

455 "ml_adjusted": True, 

456 "original_action": strategy_action, 

457 "ml_action": ml_action, 

458 "adjustment_reason": "ml_disagreement_low_confidence", 

459 "original_confidence": strategy_confidence, 

460 "ml_confidence": ml_confidence, 

461 }, 

462 ) 

463 else: 

464 # ML信頼度が高くない場合は加重平均のみ 

465 adjusted_confidence = base_confidence 

466 self.logger.debug(f"📊 ML通常統合（加重平均のみ）: {base_confidence:.3f}") 

467 

468 # 統合結果を新しいStrategySignalオブジェクトとして返す 

469 from ...strategies.base.strategy_base import StrategySignal 

470 

471 return StrategySignal( 

472 strategy_name=strategy_signal.strategy_name, 

473 timestamp=strategy_signal.timestamp, 

474 action=strategy_action, # アクションは戦略のものを維持 

475 confidence=adjusted_confidence, # ML統合後の信頼度 

476 strength=adjusted_confidence, # strengthも更新 

477 current_price=strategy_signal.current_price, 

478 entry_price=strategy_signal.entry_price, 

479 stop_loss=strategy_signal.stop_loss, 

480 take_profit=strategy_signal.take_profit, 

481 position_size=strategy_signal.position_size, 

482 risk_ratio=strategy_signal.risk_ratio, 

483 indicators=strategy_signal.indicators, 

484 reason=strategy_signal.reason, 

485 metadata={ 

486 **(strategy_signal.metadata or {}), 

487 "ml_adjusted": True, 

488 "original_confidence": strategy_confidence, 

489 "ml_confidence": ml_confidence, 

490 "ml_action": ml_action, 

491 "is_agreement": is_agreement, 

492 }, 

493 ) 

494 

495 except Exception as e: 

496 self.logger.error(f"ML統合エラー: {e} - 戦略シグナルのみ使用") 

497 return strategy_signal

499 async def _execute_approved_trades(self, trade_evaluation, cycle_id): 

500 """Phase 8a: 承認された取引の実行（Silent Failure修正済み）""" 

501 try: 

502 execution_result = None 

503 # Enum比較を正しく実装（str変換問題解決・循環import回避） 

504 decision_value = getattr(trade_evaluation, "decision", None) 

505 # RiskDecision.APPROVEDの値は"approved"なので文字列比較で回避 

506 if decision_value == "approved" or ( 

507 hasattr(decision_value, "value") and decision_value.value == "approved" 

508 ): 

509 self.logger.debug( 

510 f"取引実行開始 - サイクル: {cycle_id}, アクション: {getattr(trade_evaluation, 'side', 'unknown')}" 

511 ) 

512

516 )

522

524 trade_evaluation 

525 ) 

526 

527 self.logger.info( 

528 f"✅ 取引実行完了 - サイクル: {cycle_id}, 結果: {execution_result.success if execution_result else 'None'}" 

529 ) 

530 

531 await self.orchestrator.trading_logger.log_execution_result( 

532 execution_result, cycle_id 

533 ) 

534 else: 

535 # holdシグナルや取引拒否の詳細説明 

536 decision = getattr(trade_evaluation, "decision", "unknown") 

537 side = getattr(trade_evaluation, "side", "unknown") 

538 reason = getattr(trade_evaluation, "denial_reasons", ["理由不明"]) 

539 

540 if side.lower() in ["hold", "none"]: 

541 self.logger.info( 

542 f"📤 holdシグナル処理 - サイクル: {cycle_id}, アクション: {side}, 判定: {decision}" 

543 ) 

544 else: 

545 self.logger.debug( 

546 f"取引未承認 - サイクル: {cycle_id}, 決定: {decision}, アクション: {side}, 理由: {reason}" 

547 ) 

548 await self.orchestrator.trading_logger.log_trade_decision( 

549 trade_evaluation, cycle_id 

550 ) 

551 except AttributeError as e: 

552 # ExecutionServiceにexecute_tradeメソッドがない場合の詳細エラー 

553 self.logger.error(f"❌ ExecutionServiceメソッドエラー - サイクル: {cycle_id}: {e}")

638 async def _pre_execution_verification(self, trade_evaluation, cycle_id) -> dict: 

639 """ 

640 取引実行直前最終検証（口座残高使い切り防止・追加安全チェック） 

641 

642 リスク管理後から実際の取引実行までの間に状況が変わった可能性を考慮した 

643 最後の安全チェック。二重チェックによる安全性向上。 

644 

645 Args: 

646 trade_evaluation: 取引評価結果 

647 cycle_id: 取引サイクルID 

648 

649 Returns: 

650 Dict: {"allowed": bool, "reason": str, "additional_info": dict} 

651 """ 

652 try: 

653 self.logger.debug(f"🔍 取引直前最終検証開始 - サイクル: {cycle_id}") 

654 

655 # 1. 基本情報確認 

656 side = getattr(trade_evaluation, "side", "unknown") 

657 position_size = getattr(trade_evaluation, "position_size", 0.0) 

658 

659 if side.lower() in ["hold", "none", ""] or position_size <= 0: 

660 return {"allowed": False, "reason": "holdシグナルまたは無効なポジションサイズ"} 

661 

662 # 2. ExecutionService経由でポジション制限再確認 

663 try: 

664 execution_service = self.orchestrator.execution_service 

665 if hasattr(execution_service, "_check_position_limits"): 

666 position_check = execution_service._check_position_limits(trade_evaluation) 

667 if not position_check.get("allowed", True): 

668 return { 

669 "allowed": False, 

670 "reason": f"ポジション制限再確認失敗: {position_check.get('reason', '不明')}", 

671 } 

672 except Exception as e: 

673 self.logger.warning(f"⚠️ ポジション制限再確認エラー: {e}") 

674 

675 # 3. 現在残高確認（最新残高でのチェック） 

676 try: 

677 current_balance = await self._get_current_balance() 

678 if current_balance is not None and current_balance > 0: 

679 # 最小取引可能残高チェック 

680 min_trade_size = get_threshold("trading.min_trade_size", 0.0001) 

681 estimated_cost = min_trade_size * 16700000 # BTC価格概算 

682 

683 if current_balance < estimated_cost * 1.5: # 1.5倍の余裕を確保 

684 return { 

685 "allowed": False, 

686 "reason": f"残高不足: ¥{current_balance:,.0f} < 必要残高¥{estimated_cost * 1.5:,.0f}", 

687 } 

688 except Exception as e: 

689 self.logger.warning(f"⚠️ 現在残高確認エラー: {e}") 

690 

691 # 4. 市場急変状況チェック 

692 try: 

693 # 直近の価格ボラティリティ確認（簡易実装） 

694 market_volatility_check = await self._check_current_market_volatility() 

695 if market_volatility_check and not market_volatility_check.get("stable", True): 

696 volatile_threshold = get_threshold( 

697 "trading.anomaly.max_volatility_for_trade", 0.05 

698 ) 

699 current_volatility = market_volatility_check.get("volatility", 0.0) 

700 

701 if current_volatility > volatile_threshold: 

702 return { 

703 "allowed": False, 

704 "reason": f"市場急変検出: ボラティリティ {current_volatility * 100:.1f}% > {volatile_threshold * 100:.0f}%", 

705 } 

706 except Exception as e: 

707 self.logger.debug(f"市場ボラティリティチェックエラー（非致命的）: {e}") 

708 

709 # 5. 緊急時条件確認（Emergency Stop-Lossがアクティブでないか） 

710 try: 

711 if hasattr(self.orchestrator, "execution_service"): 

712 emergency_status = await self._check_emergency_conditions() 

713 if emergency_status and not emergency_status.get("normal", True): 

714 return { 

715 "allowed": False, 

716 "reason": f"緊急時条件検出: {emergency_status.get('reason', '不明な緊急事態')}", 

717 } 

718 except Exception as e: 

719 self.logger.debug(f"緊急時条件チェックエラー（非致命的）: {e}") 

720 

721 # 6. システム健全性最終確認 

722 health_status = await self._check_system_health_for_trading() 

723 if not health_status.get("healthy", False): 

724 return { 

725 "allowed": False, 

726 "reason": f"システム健全性問題: {health_status.get('issue', '不明')}", 

727 } 

728 

729 self.logger.debug(f"✅ 取引直前最終検証通過 - サイクル: {cycle_id}") 

730 

731 return { 

732 "allowed": True, 

733 "reason": "全検証通過", 

734 "additional_info": { 

735 "verification_time": datetime.now(), 

736 "checks_performed": [ 

737 "position_limits", 

738 "balance", 

739 "volatility", 

740 "emergency", 

741 "health", 

742 ], 

743 }, 

744 } 

745 

746 except Exception as e: 

747 self.logger.error(f"❌ 取引直前検証エラー - サイクル: {cycle_id}: {e}") 

748 return {"allowed": False, "reason": f"検証処理エラー: {e}"}

780 """取引実行可能システム健全性チェック""" 

781 try: 

782 # 基本的なシステムコンポーネント確認 

783 checks = { 

784 "execution_service": ( 

785 hasattr(self.orchestrator, "execution_service") 

786 and self.orchestrator.execution_service is not None 

787 ), 

788 "risk_manager": ( 

789 hasattr(self.orchestrator, "risk_service") 

790 and self.orchestrator.risk_service is not None 

791 ), 

792 "data_service": ( 

793 hasattr(self.orchestrator, "data_service") 

794 and self.orchestrator.data_service is not None 

795 ), 

796 } 

797 

798 if all(checks.values()): 

799 return {"healthy": True} 

800 else: 

801 failed_components = [comp for comp, status in checks.items() if not status] 

802 return { 

803 "healthy": False, 

804 "issue": f"必須コンポーネント異常: {', '.join(failed_components)}", 

805 }

808 return {"healthy": False, "issue": f"ヘルスチェック例外: {e}"}
