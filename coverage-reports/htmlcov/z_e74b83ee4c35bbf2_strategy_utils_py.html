69 @staticmethod 

70 def _extract_15m_atr( 

71 df: pd.DataFrame, multi_timeframe_data: Optional[Dict[str, pd.DataFrame]] = None 

72 ) -> Optional[float]: 

73 """ 

74 Phase 31: 15m足ATR優先取得・4h足ATRフォールバック 

75 

76 Args: 

77 df: メインタイムフレームデータ（4h足） 

78 multi_timeframe_data: マルチタイムフレームデータ 

79 

80 Returns: 

81 ATR値（15m優先、なければ4h、取得失敗ならNone） 

82 """ 

83 logger = get_logger() 

84 

85 # Phase 31: 15m足ATR優先取得 

86 if multi_timeframe_data and "15m" in multi_timeframe_data: 

87 try: 

88 df_15m = multi_timeframe_data["15m"] 

89 if "atr_14" in df_15m.columns and len(df_15m) > 0: 

90 atr_15m = float(df_15m["atr_14"].iloc[-1]) 

91 if atr_15m > 0: 

92 logger.info(f"✅ Phase 31: 15m足ATR使用 = {atr_15m:.0f}円") 

93 return atr_15m 

94 except Exception as e: 

95 logger.warning(f"15m足ATR取得失敗: {e}") 

96 

97 # フォールバック: 4h足ATR取得 

98 try: 

99 if "atr_14" in df.columns and len(df) > 0: 

100 atr_4h = float(df["atr_14"].iloc[-1]) 

101 if atr_4h > 0: 

102 logger.info(f"⚠️ Phase 31フォールバック: 4h足ATR使用 = {atr_4h:.0f}円") 

103 return atr_4h 

104 except Exception as e: 

105 logger.error(f"4h足ATR取得失敗: {e}") 

106 

107 return None

109 @staticmethod 

110 def _calculate_adaptive_atr_multiplier( 

111 current_atr: float, atr_history: Optional[List[float]] = None 

112 ) -> float: 

113 """ 

114 Phase 30: 適応型ATR倍率計算 

115 

116 Args: 

117 current_atr: 現在のATR値 

118 atr_history: ATR履歴（ボラティリティ判定用、Noneの場合はデフォルト倍率） 

119 

120 Returns: 

121 適応型ATR倍率 

122 """ 

123 from ...core.config import get_threshold 

124 

125 # 適応型ATR機能が無効な場合 

126 if not get_threshold("position_management.stop_loss.adaptive_atr.enabled", True): 

127 return get_threshold("position_management.stop_loss.default_atr_multiplier", 2.0) 

128 

129 # ATR履歴がない場合はデフォルト 

130 if not atr_history or len(atr_history) < 10: 

131 return get_threshold( 

132 "position_management.stop_loss.adaptive_atr.normal_volatility.multiplier", 2.0 

133 ) 

134 

135 # ATR平均計算 

136 import numpy as np 

137 

138 avg_atr = np.mean(atr_history) 

139 

140 # ボラティリティ状態判定 

141 low_threshold = get_threshold( 

142 "position_management.stop_loss.adaptive_atr.low_volatility.threshold_ratio", 0.7 

143 ) 

144 high_threshold = get_threshold( 

145 "position_management.stop_loss.adaptive_atr.high_volatility.threshold_ratio", 1.3 

146 ) 

147 

148 volatility_ratio = current_atr / avg_atr if avg_atr > 0 else 1.0 

149 

150 # ボラティリティに応じた倍率選択 

151 if volatility_ratio < low_threshold: 

152 # 低ボラティリティ → 広めのSL 

153 return get_threshold( 

154 "position_management.stop_loss.adaptive_atr.low_volatility.multiplier", 2.5 

155 ) 

156 elif volatility_ratio > high_threshold: 

157 # 高ボラティリティ → 狭めのSL（急変時対策） 

158 return get_threshold( 

159 "position_management.stop_loss.adaptive_atr.high_volatility.multiplier", 1.5 

160 ) 

161 else: 

162 # 通常ボラティリティ → 標準SL 

163 return get_threshold( 

164 "position_management.stop_loss.adaptive_atr.normal_volatility.multiplier", 2.0 

165 )

167 @staticmethod 

168 def calculate_stop_loss_take_profit( 

169 action: str, 

170 current_price: float, 

171 current_atr: float, 

172 config: Dict[str, Any], 

173 atr_history: Optional[List[float]] = None, 

174 ) -> Tuple[Optional[float], Optional[float]]: 

175 """ 

176 ストップロス・テイクプロフィット計算 

177 

178 Args: 

179 action: エントリーアクション（buy/sell） 

180 current_price: 現在価格 

181 current_atr: 現在のATR値（Phase 30: 15分足ATR推奨） 

182 config: 戦略設定（stop_loss_atr_multiplier, take_profit_ratio含む） 

183 atr_history: ATR履歴（Phase 30: 適応型ATR用） 

184 

185 Returns: 

186 (stop_loss, take_profit)のタプル 

187 """ 

188 logger = get_logger() 

189 from ...core.config import get_threshold 

190 

191 try: 

192 if action not in [EntryAction.BUY, EntryAction.SELL]: 

193 return None, None 

194 

195 # Phase 30: 適応型ATR倍率計算 

196 stop_loss_multiplier = RiskManager._calculate_adaptive_atr_multiplier( 

197 current_atr, atr_history 

198 ) 

199 

200 take_profit_ratio = config.get( 

201 "take_profit_ratio", DEFAULT_RISK_PARAMS["take_profit_ratio"] 

202 ) 

203 min_atr = config.get("min_atr_threshold", DEFAULT_RISK_PARAMS["min_atr_threshold"]) 

204 

205 # ATR最小値チェック（ゼロ除算回避） 

206 if current_atr < min_atr: 

207 logger.warning(f"ATR値が小さすぎます: {current_atr:.6f} < {min_atr}") 

208 current_atr = min_atr 

209 

210 # ストップロス距離計算 

211 stop_loss_distance = current_atr * stop_loss_multiplier 

212 

213 # Phase 30: 最小SL距離保証 

214 # Phase 32.1修正: ATR優先・動的保証実現 

215 min_distance_enabled = get_threshold( 

216 "position_management.stop_loss.min_distance.enabled", True 

217 ) 

218 if min_distance_enabled: 

219 override_atr = get_threshold( 

220 "position_management.stop_loss.min_distance.override_atr", False 

221 ) 

222 

223 if override_atr: 

224 # 旧動作: 固定1%保証がATRを上書き 

225 min_distance_ratio = get_threshold( 

226 "position_management.stop_loss.min_distance.ratio", 0.01 

227 ) 

228 min_sl_distance = current_price * min_distance_ratio 

229 

230 if stop_loss_distance < min_sl_distance: 

231 logger.info( 

232 f"📏 固定1%SL距離保証適用: {stop_loss_distance:.0f}円 → {min_sl_distance:.0f}円" 

233 ) 

234 stop_loss_distance = min_sl_distance 

235 else: 

236 # Phase 32.1新動作: ATR×倍率を最小値として保証（動的保証） 

237 min_atr_multiplier = get_threshold( 

238 "position_management.stop_loss.min_distance.min_atr_multiplier", 1.5 

239 ) 

240 min_atr_based = current_atr * min_atr_multiplier 

241 

242 # ATRベース最小値と計算値を比較 

243 if stop_loss_distance < min_atr_based: 

244 logger.info( 

245 f"📏 ATRベース動的SL保証適用: {stop_loss_distance:.0f}円 → {min_atr_based:.0f}円 " 

246 f"(ATR {current_atr:.0f}円 × {min_atr_multiplier:.1f}倍)" 

247 ) 

248 stop_loss_distance = min_atr_based 

249 

250 # Phase 32.1修正: ATR極小時のみ固定1%フォールバック適用 

251 # ATRベース最小保証より小さい場合のみ固定1%を適用 

252 min_distance_ratio = get_threshold( 

253 "position_management.stop_loss.min_distance.ratio", 0.01 

254 ) 

255 absolute_min = current_price * min_distance_ratio 

256 

257 # ATRベース最小保証（min_atr_based）と固定1%を比較し、 

258 # ATRベース最小保証より計算値が小さい場合のみ固定1%適用 

259 if stop_loss_distance < min_atr_based and stop_loss_distance < absolute_min: 

260 logger.warning( 

261 f"⚠️ ATR極小・固定1%フォールバック適用: {stop_loss_distance:.0f}円 → {absolute_min:.0f}円 " 

262 f"(ATR {current_atr:.0f}円が極小のため)" 

263 ) 

264 stop_loss_distance = absolute_min 

265 

266 # BUY/SELL別の計算 

267 if action == EntryAction.BUY: 

268 stop_loss = current_price - stop_loss_distance 

269 take_profit = current_price + (stop_loss_distance * take_profit_ratio) 

270 else: # SELL 

271 stop_loss = current_price + stop_loss_distance 

272 take_profit = current_price - (stop_loss_distance * take_profit_ratio) 

273 

274 # 値の妥当性確認 

275 if stop_loss <= 0 or take_profit <= 0: 

276 logger.error( 

277 f"無効なストップロス/テイクプロフィット: SL={stop_loss:.2f}, TP={take_profit:.2f}" 

278 ) 

279 return None, None 

280 

281 logger.info( 

282 f"🎯 Phase 30 SL/TP計算: ATR={current_atr:.0f}円, 倍率={stop_loss_multiplier:.2f}, " 

283 f"SL距離={stop_loss_distance:.0f}円（{stop_loss_distance / current_price * 100:.2f}%）" 

284 ) 

285 

286 return stop_loss, take_profit 

287 

288 except Exception as e: 

289 logger.error(f"ストップロス・テイクプロフィット計算エラー: {e}") 

290 return None, None

292 @staticmethod 

293 def calculate_position_size(confidence: float, config: Dict[str, Any]) -> float: 

294 """ 

295 ポジションサイズ計算 

296 

297 Args: 

298 confidence: シグナル信頼度（0.0-1.0） 

299 config: 戦略設定（position_size_base含む） 

300 

301 Returns: 

302 計算されたポジションサイズ 

303 """ 

304 logger = get_logger() 

305 

306 try: 

307 # パラメータ取得 

308 base_size = config.get("position_size_base", DEFAULT_RISK_PARAMS["position_size_base"]) 

309 max_size = config.get("max_position_size", DEFAULT_RISK_PARAMS["max_position_size"]) 

310 

311 # 信頼度による調整 

312 position_size = base_size * max(0.0, min(1.0, confidence)) 

313 

314 # 最大サイズ制限 

315 position_size = min(position_size, max_size) 

316 

317 return position_size 

318 

319 except Exception as e: 

320 logger.error(f"ポジションサイズ計算エラー: {e}") 

321 return 0.0

323 @staticmethod 

324 def calculate_risk_ratio(current_price: float, stop_loss: Optional[float]) -> Optional[float]: 

325 """ 

326 リスク比率計算 

327 

328 Args: 

329 current_price: 現在価格 

330 stop_loss: ストップロス価格 

331 

332 Returns: 

333 リスク比率（None if 計算不可） 

334 """ 

335 try: 

336 if stop_loss is None or current_price <= 0: 

337 return None 

338 

339 risk_ratio = abs(stop_loss - current_price) / current_price 

340 return risk_ratio 

341 

342 except Exception: 

343 return None

395 @staticmethod 

396 def create_signal_with_risk_management( 

397 strategy_name: str, 

398 decision: Dict[str, Any], 

399 current_price: float, 

400 df: pd.DataFrame, 

401 config: Dict[str, Any], 

402 strategy_type: Optional[str] = None, 

403 multi_timeframe_data: Optional[Dict[str, pd.DataFrame]] = None, 

404 ) -> StrategySignal: 

405 """ 

406 リスク管理付きシグナル生成 

407 

408 Args: 

409 strategy_name: 戦略名 

410 decision: 戦略の判定結果（action, confidence, strength等） 

411 current_price: 現在価格 

412 df: 市場データ（ATR計算用） 

413 config: 戦略設定 

414 strategy_type: 戦略タイプ（メタデータ用） 

415 multi_timeframe_data: マルチタイムフレームデータ（Phase 31対応） 

416 

417 Returns: 

418 完全なStrategySignal

421 

422 try: 

423 action = decision.get("action", EntryAction.HOLD) 

424 confidence = decision.get("confidence", 0.0) 

425 strength = decision.get("strength", 0.0) 

426 reason = decision.get("analysis", decision.get("reason", "")) 

427 

428 # リスク管理計算（エントリーシグナルのみ） 

429 stop_loss = None 

430 take_profit = None 

431 position_size = None 

432 risk_ratio = None 

433 

434 if action in [EntryAction.BUY, EntryAction.SELL]: 

435 # Phase 31: 15m足ATR優先取得 

436 current_atr = RiskManager._extract_15m_atr(df, multi_timeframe_data) 

437 if current_atr is None: 

438 logger.warning(f"ATR取得失敗: {strategy_name}") 

439 return SignalBuilder._create_error_signal( 

440 strategy_name, current_price, "ATR取得失敗" 

441 ) 

442 

443 # Phase 30: ATR履歴取得（適応型ATR用） 

444 atr_history = None 

445 if multi_timeframe_data and "15m" in multi_timeframe_data: 

446 try: 

447 df_15m = multi_timeframe_data["15m"] 

448 if "atr_14" in df_15m.columns: 

449 atr_history = df_15m["atr_14"].dropna().tail(20).tolist() 

450 except Exception: 

451 pass 

452 

453 # ストップロス・テイクプロフィット計算 

454 stop_loss, take_profit = RiskManager.calculate_stop_loss_take_profit(

456 )

459 position_size = RiskManager.calculate_position_size(confidence, config) 

460 

461 # リスク比率計算 

462 risk_ratio = RiskManager.calculate_risk_ratio(current_price, stop_loss) 

463 

464 # メタデータ作成 

465 metadata = { 

466 "strategy_type": strategy_type, 

467 "risk_calculated": action in [EntryAction.BUY, EntryAction.SELL], 

468 "decision_metadata": decision.get("metadata", {}), 

469 } 

470 

471 # StrategySignal生成 

472 return StrategySignal( 

473 strategy_name=strategy_name, 

474 timestamp=datetime.now(), 

475 action=action, 

476 confidence=confidence, 

477 strength=strength, 

478 current_price=current_price, 

479 entry_price=current_price, 

480 stop_loss=stop_loss, 

481 take_profit=take_profit, 

482 position_size=position_size, 

483 risk_ratio=risk_ratio, 

484 reason=reason, 

485 metadata=metadata, 

486 ) 

487 

488 except Exception as e: 

489 logger.error(f"シグナル生成エラー ({strategy_name}): {e}") 

490 return SignalBuilder._create_error_signal( 

491 strategy_name, current_price, f"シグナル生成エラー: {e}" 

492 )
